ޙަދީޘްއެމްވީ - ނަވަވީގެ 40 ޙަދީޘްގެ


#01

[إِنَّمَا الْأَعْمَالُ بِالنِّيَّاتِ]

[ޢަމަލުތައް ބިނާވެގެން ވަނީ ނިޔަތްތަކުގެ މައްޗަށެވެ]

عَنْ أَمِيرِ الْمُؤْمِنِينَ، أَبِي حَفْصٍ -عُمَرَ بْنِ الْخَطَّابِ- رَضِيَ اللَّهُ عَنْهُ قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «إِنَّمَا الْأَعْمَالُ بِالنِّيَّاتِ، وَإِنَّمَا لِكُلِّ امْرِئٍ مَا نَوَى، فَمَنْ كَانَتْ هِجْرَتُهُ إِلَى اللَّهِ وَرَسُولِهِ، فَهِجْرَتُهُ إِلَى اللَّهِ وَرَسُولِهِ، وَمَنْ كَانَتْ هِجْرَتُهُ لِدُنْيَا يُصِيبُهَا، أَوِ امْرَأَةٍ يَنْكِحُهَا، فَهِجْرَتُهُ إِلَى مَا هَاجَرَ إِلَيهِ»

އަމީރުލް މުއުމިނީން، އަބޫ ޙަފްޞު، ޢުމަރު ބިން އަލްޚައްޠާބު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”ނިޔަތްތަކުގެ މައްޗަށް މެނުވީ ޢަމަލުތައް ބިނާ ވެގެނެއް ނުވެއެވެ. އަދި ކޮންމެ މީހަކަށް ވެސް ވަނީ، ހަމައެކަނި އެ މީހަކު ނިޔަތް ގަތް ކަމެކެވެ. ވީމާ، މީހަކު ހިޖުރަ ކުރީ ﷲ އާއި އެ އިލާހުގެ ރަސޫލާއަށްޓަކައި ނަމަ، ފަހެ އޭނާގެ ހިޖުރަ ވަނީ ﷲ އާއި އެ އިލާހުގެ ރަސޫލާއަށެވެ. އަދި މީހަކު ހިޖުރަ ކުރީ ދުނިޔެ ހޯދުމަށް ނުވަތަ އަންހެނަކާ އިނުމަށް ކަމުގައި ވަނީ ނަމަ، ފަހެ އެ މީހެއްގެ ހިޖުރަ ވަނީ، އޭނާ އެ ހިޖުރަ ކުރި ކަމަކަށެވެ.“

رَوَاهُ إِمَامَا الْمُحَدِّثِينَ: أَبُو عَبْدِ اللَّهِ مُحَمَّدُ بْنُ إِسْمَاعِيلَ بْنِ إِبْرَاهِيمَ بْنِ الْمُغِيرَةِ بْنِ بَرْدِزْبَهْ الْبُخَارِيُّ، وَأَبُو الْحُسَينِ مُسْلِمُ بْنُ الْحَجَّاجِ بْنِ مُسْلِمٍ الْقُشَيرِيُّ النَّيسَابُورِيُّ رَضِيَ اللَّهُ عَنْهُمَا فِي صَحِيحَيهِمَا -اللَّذَينِ هُمَا أَصَحُّ الْكُتُبِ الْمُصَنَّفَةِ-. [الْبُخَارِيُّ 0001، وَمُسْلِمٌ 1907]

ރިވާ ކުރެއްވީ ޙަދީޘް ޢިލްމުވެރިންގެ ދެ އިމާމުން ކަމަށްވާ: އަބޫ ޢަބްދު ﷲ މުޙައްމަދު ބިން އިސްމާޢީލު ބިން އިބްރާހީމް ބިން އަލްމުޣީރާ ބިން ބަރްދިޒްބަހް އަލްބުޚާރީއާއި، އަބުލް ޙުސައިން މުސްލިމު ބިން އަލްޙައްޖާޖު ބިން މުސްލިމު އަލްޤުޝައިރީ އައްނައިސާޕޫރީ، މި ދެ އިމާމުންގެ ދެ ޞަޙީޙުގައެވެ. ﷲ އެ ދެ އިމާމުންނަށް ރުއްސުން ލައްވާށެވެ. އެއީ (އިންސާނުން ލިޔުނު) ފޮތްތަކުގެ ތެރެއިން ލިޔެވިފައިވާ އެންމެ ޞައްޙަ ދެ ފޮތެވެ. [ބުޚާރީ 0001، މުސްލިމު 1907]

#02

[فَإِنَّهُ جِبْرِيلُ أَتَاكُمْ يُعَلِّمُكُمْ دِينَكُمْ]

[ޖިބްރީލުގެފާނު އެ ވަޑައިގަތީ، ތިޔަބައިމީހުންގެ ދީން ތިޔަބައިމީހުންނަށް އުނގަންނަވައި ދެއްވުމަށްޓަކައެވެ]

عَنْ عُمَرَ رَضِيَ اللَّهُ عَنْهُ -أَيضًا- قَالَ: بَينَمَا نَحْنُ عِنْدَ رَسُولِ اللَّهِ ﷺ ذَاتَ يَومٍ، إِذْ طَلَعَ عَلَينَا رَجُلٌ، شَدِيدُ بَيَاضِ الثِّيَابِ، شَدِيدُ سَوَادِ الشَّعْرِ، لَا يُرَى عَلَيهِ أَثَرُ السَّفَرِ، وَلَا يَعْرِفُهُ مِنَّا أَحَدٌ، حَتَّى جَلَسَ إِلَى النَّبِيِّ ﷺ، فَأَسْنَدَ رُكْبَتَيهِ إِلَى رُكْبَتَيهِ، وَوَضَعَ كَفَّيهِ عَلَى فَخِذَيهِ، وَقَالَ: يَا مُحَمَّدُ! أَخْبِرْنِي عَنْ الْإِسْلَامِ؟ فَقَالَ رَسُولُ اللَّهِ ﷺ: «الْإِسْلَامُ: أَنْ تَشْهَدَ أَلَّا إِلَهَ إِلَّا اللَّهُ وَأَنَّ مُحَمَّدًا رَسُولُ اللَّهِ، وَتُقِيمَ الصَّلَاةَ، وَتُؤْتِيَ الزَّكَاةَ، وَتَصُومَ رَمَضَانَ، وَتَحُجَّ الْبَيتَ إِنِ اسْتَطَعْتَ إِلَيهِ سَبِيلًا» قَالَ: صَدَقْتَ -فَعَجِبْنَا لَهُ، يَسْأَلُهُ وَيُصَدِّقُهُ-. قَالَ: فَأَخْبِرْنِي عَنِ الْإِيمَانِ؟ قَالَ: «أَنْ تُؤْمِنَ بِاللَّهِ، وَمَلَائِكَتِهِ، وَكُتُبِهِ، وَرُسُلِهِ، وَالْيَومِ الْآخِرِ، وَتُؤْمِنَ بِالْقَدَرِ خَيرِهِ وَشَرِّهِ» قَالَ: صَدَقْتَ. قَالَ: فَأَخْبِرْنِي عَنِ الْإِحْسَانِ؟ قَالَ: «أَنْ تَعْبُدَ اللَّهَ كَأَنَّكَ تَرَاهُ، فَإِنْ لَمْ تَكُنْ تَرَاهُ فَإِنَّهُ يَرَاكَ» قَالَ: فَأَخْبِرْنِي عَنِ السَّاعَةِ؟ قَالَ: «مَا الْمَسْئُولُ عَنْهَا بِأَعْلَمَ مِنَ السَّائِلِ» قَالَ: فَأَخْبِرْنِي عَنْ أَمَارَاتِهَا؟ قَالَ: «أَنْ تَلِدَ الْأَمَةُ رَبَّتَهَا، وَأَنْ تَرَى الْحُفَاةَ الْعُرَاةَ الْعَالَةَ رِعَاءَ الشَّاءِ، يَتَطَاوَلُونَ فِي الْبُنْيَانِ» قَالَ: ثُمَّ انْطَلَقَ فَلَبِثْتُ مَلِيًّا، ثُمَّ قَالَ لِي: «يَا عُمَرُ! أَتَدْرِي مَنِ السَّائِلُ؟» قُلْتُ: اللَّهُ وَرَسُولُهُ أَعْلَمُ. قَالَ: «فَإِنَّهُ جِبْرِيلُ أَتَاكُمْ يُعَلِّمُكُمْ دِينَكُمْ»

އަދިވެސް ޢުމަރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ އަރިހުގައި ތިމަންމެން ވަނިކޮށް، އެ ތަނަށް މީހަކު އަތުވެއްޖެއެވެ. އެ މީހާގެ ފޭރާމުގެ ހުދުކަން ނުހަނު ގަދައެވެ. އަދި އިސްތަށީގެ ކަޅުކަން ވެސް އެހާމެ ގަދައެވެ. ދަތުރެއްކޮށްފައި އައިކަމުގެ އެއްވެސް އަސަރެއް އެ މީހާގެ ކިބައިން ފެންނާކަށް ނެތެވެ. އަދި ތިމަންމެން ކުރެ އެކަކަށް ވެސް އެއީ ކާކު ކަމެއް ނޭނގެއެވެ. އެ މީހާ އައިސް، ނަބިއްޔާ ﷺ ގެ ކަކޫފުޅާއި އެ މީހާގެ ކަކުލާ ދިމާވާ ގޮތަށް އިށީނދެ، ދެ އަތްތިލަ ފަލަމަސްގަނޑުގެ މަތީ ބާއްވާފައި ދެންނެވިއެވެ: އޭ މުޙައްމަދުގެފާނެވެ. އިސްލާމްކަމާ ބެހޭ ގޮތުން ތިމަންނާއަށް ޚަބަރު ދެއްވާށެވެ. އެ ހިނދު ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”އިސްލާމްކަމަކީ، ﷲ ފިޔަވައި (އަޅުކަން ޙައްޤުވާ) އެހެން އެއްވެސް އިލާހަކު ނުވާ ކަމަށާއި، މުޙައްމަދުގެފާނީ ﷲގެ ރަސޫލާ ކަމަށް ތިބާ ހެކިވުމެވެ. އަދި ނަމާދު ގާއިމުކޮށް، (މުދަލު) ޒަކާތް ދީ، ރަމަޟާން މަހު ރޯދަ ހިފައި، އަދި ތިބާއަށް ކުޅަދާނަކަން ލިބި އެ ކަމަށް މަގުފަހިވެއްޖެ ނަމަ، ގެފުޅަށް ގޮސް ޙައްޖުވުމެވެ.“ އޭނާ ދެންނެވިއެވެ: ތިޔަ ވިދާޅުވީ ތެދުފުޅެކެވެ. -ޙަދީޘް ކިޔައިދެއްވަމުން ޢުމަރުގެފާނު ދެންނެވިއެވެ: އެ މީހާއާމެދު ތިމަންމެން އަޖައިބު ވީމެވެ. (އެހެނީ) އެ މީހާ ނަބިއްޔާއާ ސުވާލުކޮށްފައި، ޚުދު އަމިއްލައަށް ނަބިއްޔާގެ ބަސްފުޅު ވެސް ތެދު ކުރެއެވެ- ދެން އެ މީހާ ދެންނެވިއެވެ: އީމާންކަމާ ބެހޭ ގޮތުން ތިމަންނާއަށް ޚަބަރު ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެއީ ﷲ އަށާއި، އެ އިލާހުގެ މަލާއިކަތުންނަށާއި، އެ އިލާހުގެ ފޮތްތަކަށާއި، އެ އިލާހުގެ ރަސޫލުންނަށާއި، އާޚިރަތް ދުވަހަށް ތިބާ އީމާން ވުމެވެ. އަދި ޤަޟާޤަދަރު، އޭގެ ހެޔޮ މިންވަރާއި ނުބައި މިންވަރަށް ތިބާ އީމާން ވުމެވެ.“ އެ މީހާ ދެންނެވިއެވެ: ތިޔަ ވިދާޅުވީ ތެދުފުޅެކެވެ. އިޙުސާންކަމާ ބެހޭ ގޮތުން ތިމަންނާއަށް ޚަބަރު ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެއީ ތިބާއަށް ﷲ ފެނިހުރެ އަޅުކަން ކުރާނެ ފަދައިން އެ އިލާއަށް އަޅުކަން ކުރުމެވެ. ފަހެ ތިބާއަށް އެ އިލާހު ނުފެނުނު ކަމުގައި ވިޔަސް، ހަމަކަށަވަރުން އެ އިލާހު ތިބާއަށް ބައްލަވާ ވޮޑިގެންވެއެވެ.“ ދެން އޭނާ ދެންނެވިއެވެ: ޤިޔާމަތް ވާނެ ވަގުތަކާ ބެހޭ ގޮތުން ތިމަންނާއަށް ޚަބަރު ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެކަމާ ބެހޭ ގޮތުން، ސުވާލު ކުރި ފަރާތަށް ވުރެ ބޮޑަށް، ސުވާލު ކުރެވުނު ފަރާތަށް އެނގިގެނެއް ނުވެއެވެ.“ (އެބަހީ: ދެ މީހުން ކުރެ އެކަކަށް ވެސް އެނގިގެނެއް ނުވެއެވެ.) ދެން އެ މީހާ ދެންނެވިއެވެ: އޭގެ ޢަލާމާތްތަކާ ބެހޭ ގޮތުން ތިމަންނާއަށް ޚަބަރު ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެއީ އަޅު އަންހެނަކު އަންހެން ދަރިއެއް ވިހައިގެން، (އެ ލިބުނު ދަރި) އެ އަޅު އަންހެނާގެ ސާހިބާއަށް ވުމެވެ. އަދި ހުސްފަޔާ އޮރިޔާންކަންމަތީ ފަޤީރު ޙާލުގައި ބަކަރި ހުއިހައްޕާ އުޅުނު މީހުން، އިމާރާތްތައް އުސް ކުރުމުގައި ވާދަކޮށް އުޅޭތަން ތިބާއަށް ފެނުމެވެ.“ ދެން ޢުމަރުގެފާނު ދެންނެވިއެވެ: އެއަށްފަހު، އެ މީހަކު ހިނގައްޖެއެވެ. ދެން ތިމަން ދުވަސްތަކެއް ވޭތު ކުރުމަށް ފަހު، ނަބިއްޔާ ތިމަންނާއަށް ޙަދީޘް ކުރެއްވިއެވެ: ”އޭ ޢުމަރުގެފާނެވެ. ސުވާލު ކުރީ ކާކުކަން އެނގިލައްވާތޯއެވެ؟“ ތިމަން ދެންނެވީމެވެ: މޮޅަށް ދެނެވޮޑިގެން ވަނީ ﷲ އެވެ. އަދި އެ އިލާހުގެ ރަސޫލާއެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން އެއީ ޖިބްރީލުގެފާނެވެ. ތިޔަބައިމީހުންގެ ގާތަށް އެ ވަޑައިގަތީ، ތިޔަބައިމީހުންގެ ދީން ތިޔަބައިމީހުންނަށް އުނގަންނަވައި ދެއްވުމަށްޓަކައެވެ.“

[بُنِيَ الْإِسْلَامُ عَلَى خَمْسٍ]

[އިސްލާމް ދީން ބިނާވެގެން ވަނީ ފަސް ކަމެއްގެ މައްޗަށެވެ]

عَنْ أَبِي عَبْدِ الرَّحْمَنِ، عَبْدِ اللَّهِ بْنِ عُمَرَ بْنِ الْخَطَّابِ رَضِيَ اللَّهُ عَنْهُمَا قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «بُنِيَ الْإِسْلَامُ عَلَى خَمْسٍ: شَهَادَةِ أَلَّا إِلَهَ إِلَّا اللَّهُ وَأَنَّ مُحَمَّدًا عَبْدُهُ وَرَسُولُهُ، وَإِقَامِ الصَّلَاةِ، وَإِيتَاءِ الزَّكَاةِ، وَحَجِّ الْبَيتِ، وَصَومِ رَمَضَانَ»

އަބޫ ޢަބްދުއްރަޙްމާނު، އަބްދު ﷲ ބިން ޢުމަރު ބިން އަލްޚައްޠާބު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”އިސްލާމް (ދީން) ބިނާވެގެން ވަނީ ފަސް ކަމެއްގެ މައްޗަށެވެ. އެއީ ﷲ ފިޔަވައި (އަޅުކަން ޙައްޤުވާ) އެހެން އެއްވެސް އިލާހަކު ނުވާކަމަށް ހެކިވުމާއި، އަދި މުޙައްމަދުގެފާނީ އެ އިލާހުގެ އަޅާ އަދި ރަސޫލާކަމަށް ހެކިވުމާއި، ނަމާދު ގާއިމު ކުރުމާއި، (މުދަލު) ޒަކާތް ދިނުމާއި، ގެފުޅަށް ގޮސް ޙައްޖުވުމާއި، ރަމަޟާން މަހުގެ ރޯދަ ހިފުމެވެ.“

#04

[إِنَّ أَحَدَكُمْ يُجْمَعُ خَلْقُهُ]

[ކޮންމެ މީހަކުމެ، ސާޅީސް ދުވަސް ވަންދެން ރަހިމުގައި އެކުވެ އުފެދިގެން އާދެއެވެ]

عَنْ أَبِي عَبْدِ الرَّحْمَنِ، عَبْدِ اللَّهِ بْنِ مَسْعُودٍ رَضِيَ اللَّهُ عَنْهُمَا قَالَ: حَدَّثَنَا رَسُولُ اللَّهِ ﷺ -وَهُوَ الصَّادِقُ الْمَصْدُوقُ-: «إِنَّ أَحَدَكُمْ يُجْمَعُ خَلْقُهُ فِي بَطْنِ أُمِّهِ أَرْبَعِينَ يَومًا، ثُمَّ يَكُونُ عَلَقَةً مِثْلَ ذَلِكَ، ثُمَّ يَكُونُ مُضْغَةً مِثْلَ ذَلِكَ. ثُمَّ يُرْسَلُ إِلَيهِ الْمَلَكُ، فَيَنْفُخُ فِيهِ الرُّوحَ، وَيُؤْمَرُ بِأَرْبَعِ كَلِمَاتٍ: بِكَتْبِ رِزْقِهِ، وَأَجَلِهِ، وَعَمَلِهِ، وَشَقِيٌّ أَو سَعِيدٌ. فَوَالَّذِي لَا إِلَهَ غَيرُهُ! إِنَّ أَحَدَكُمْ لَيَعْمَلُ بِعَمَلِ أَهْلِ الْجَنَّةِ، حَتَّى مَا يَكُونُ بَينَهُ وَبَينَهَا إِلَّا ذِرَاعٌ، فَيَسْبِقُ عَلَيهِ الْكِتَابُ، فَيَعْمَلُ بِعَمَلِ أَهْلِ النَّارِ فَيَدْخُلُهَا. وَإِنَّ أَحَدَكُمْ لَيَعْمَلُ بِعَمَلِ أَهْلِ النَّارِ، حَتَّى مَا يَكُونُ بَينَهُ وَبَينَهَا إِلَّا ذِرَاعٌ، فَيَسْبِقُ عَلَيهِ الْكِتَابُ، فَيَعْمَلُ بِعَمَلِ أَهْلِ الْجَنَّةِ فَيَدْخُلُهَا»

އަބޫ ޢަބްދުއްރަޙްމާނު، ޢަބްދު ﷲ ބިން މަސްޢޫދު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ތިމަންމެންނަށް ޙަދީޘް ކުރެއްވިއެވެ. -އަދި އެ ނަބިއްޔާ ތެދުވެރިވެފައި، (އެ ނަބިއްޔާގެ ބަސްފުޅު ވެސް) ތެދުކުރެވިގެން ވެއެވެ-: ”ހަމަކަށަވަރުން ތިޔަބައިމީހުންގެ ތެރެއިން ކޮންމެ މީހަކުމެ ސާޅީސް ދުވަސް ވަންދެން، އޭނާގެ މައިމީހާގެ ބަނޑުގައި ( އޭނާގެ މައިންބަފައިންގެ ދެ ފެން) އެކުވެ، އޭނާ އުފެދިގެން އާދެއެވެ. ދެން ގަނޑުލޭކޮޅަކަށްވެ، ހަމަ އެހައި ދުވަސް ވަންދެން އެ ގޮތުގައި އޮވެއެވެ. ދެން މަސްކޮޅަކަށްވެ، ހަމަ އެހައި ދުވަސް ވަންދެން އެ ގޮތުގައި އޮވެއެވެ. ދެން އޭނާގެ ގާތަށް މަލާއިކަތަކު ފޮނުއްވެވި، އެ މަލާއިކަތާ އޭނާއަށް ފުރާނަ އެޅުއްވުމަށް ފުމެލައްވައެވެ. އަދި އެ މަލާއިކަތާއަށް ހަތަރު ކަމަކުން އަމުރު ކުރެއްވޭނެއެވެ. އެއީ އެ މީހާގެ ރިޒުގާއި، އަޖަލާއި، ޢަމަލާއި، އަދި އޭނާއަކީ އަބާއްޖަވެރިއެއް ނުވަތަ ބާއްޖަވެރިއެއްތޯ ލިޔުއްވުމަށެވެ. ފަހެ، އެ އިލާހު ފިޔަވައި (އަޅުކަން ޙައްޤުވާ) އެހެން އެއްވެސް އިލާހަކު ނުވާ ﷲ ގަންދީ ބުނަމެވެ. ހަމަކަށަވަރުން ތިޔަބައިމީހުންގެ ތެރެއިން މީހަކު، އޭނާއާއި ސުވަރުގެއާ ދެމެދު މުށެއްގެ މިންވަރު މެނުވީ ނެތިއްޖައުމަށް ދާންދެން، ސުވަރުގޭގެ އަހުލުވެރިންގެ ޢަމަލުން ޢަމަލު ކުރަމުން ދާނެއެވެ. ދެން ފަހެ، ފޮތުގައިވާ (އެބަހީ: ލައުޙުލް މަޙްފޫޡުގައިވާ) ގޮތް އިސްވެ ހިނގައި، (އެބަހީ: އުމުރުގެ ފަހުކޮޅުގައި އޭނާ އަމިއްލައަށް ބަދަލުވެ، ނުބައިކޮށް އުޅެ) ނަރަކައިގެ އަހުލުވެރިންގެ ޢަމަލުން ޢަމަލު ކުރަމުން ގޮސް، އެންމެ ފަހުން އޭނާ ނަރަކައަށް ވަދެގެންދެއެވެ. އަދި ހަމަކަށަވަރުން ތިޔަބައިމީހުންގެ ތެރެއިން މީހަކު، އޭނާއާއި ނަރަކައާ ދެމެދު މުށެއްގެ މިންވަރު މެނުވީ ނެތިއްޖައުމަށް ދާންދެން، ނަރަކައިގެ އަހުލުވެރިންގެ ޢަމަލުން ޢަމަލު ކުރަމުން ދާނެއެވެ. ދެން ފަހެ، ފޮތުގައިވާ ގޮތް އިސްވެ ހިނގައި، (އެބަހީ: އުމުރުގެ ފަހުކޮޅުގައި އޭނާ އަމިއްލައަށް ބަދަލުވެ، ރަނގަޅަށް އުޅެ) ސުވަރުގޭގެ އަހުލުވެރިންގެ ޢަމަލުން ޢަމަލު ކުރަމުން ގޮސް، އެންމެ ފަހުން އޭނާ ސުވަރުގެއަށް ވަދެގެންދެއެވެ.“

#05

[مَنْ أَحْدَثَ فِي أَمْرِنَا هَذَا]

[މި ދީނުގައި ނުވާ ކަމެއް އުފައްދައިފި މީހާ]

عَنْ أُمِّ الْمُؤْمِنِينَ -أُمِّ عَبْدِ اللَّهِ، عَائِشَةَ- رَضِيَ اللَّهُ عَنْهَا قَالَتْ: قَالَ رَسُولُ اللَّهِ ﷺ: «مَنْ أَحْدَثَ فِي أَمْرِنَا هَذَا مَا لَيسَ مِنْهُ؛ فَهُوَ رَدٌّ» رَوَاهُ الْبُخَارِيُّ وَمُسْلِمٌ.
وَفِي رِوَايَةٍ لِمُسْلِمٍ: «مَنْ عَمِلَ عَمَلًا لَيسَ عَلَيهِ أَمْرُنَا؛ فَهُوَ رَدٌّ»

އުންމުލް މުއުމީނީން، އުންމު ޢަބްދު ﷲ، ޢާއިޝާ رَضِيَ اللَّهُ عَنْهَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”އަހަރެމެންގެ މި ދީނުގައި ނުވާ އާ ކަމެއް އުފައްދައި (ބިދުޢައެއް ގެނެސްގެންފި) މީހާ (ދަންނާށެވެ!) ފަހެ، އެ ކަމެއް ވަނީ (ބާޠިލުވެ) ރައްދު ކުރެވިގެންނެވެ.“
އަދި އިމާމު މުސްލިމުގެ ރިވާޔަތެއްގައި ވެއެވެ: ”އަހަރެމެން ކުރުމަށް ނާންގަވާ ޢަމަލެއްކޮށްފި މީހާ (ދަންނާށެވެ!) ފަހެ، އެ ކަމެއް ވަނީ (ބާޠިލުވެ) ރައްދު ކުރެވިގެންނެވެ.“

#06

[إِنَّ الْحَلَالَ بَيِّنٌ وَإِنَّ الْحَرَامَ بَيِّنٌ]

[ޙަލާލު ވަނީ ބަޔާންވެގެން އަދި ޙަރާމް ވަނީ ބަޔާންވެގެން]

عَنْ أَبِي عَبْدِ اللَّهِ، النُّعْمَانِ بْنِ بَشِيرٍ رَضِيَ اللَّهُ عَنْهُمَا قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «إِنَّ الْحَلَالَ بَيِّنٌ، وَإِنَّ الْحَرَامَ بَيِّنٌ، وَبَينَهُمَا مُشْتَبِهَاتٌ لَا يَعْلَمُهُنَّ كَثِيرٌ مِنَ النَّاسِ. فَمَنِ اتَّقَى الشُّبُهَاتِ اسْتَبْرَأَ لِدِينِهِ وَعِرْضِهِ. وَمَنْ وَقَعَ فِي الشُّبُهَاتِ وَقَعَ فِي الْحَرَامِ؛ كَالرَّاعِي يَرْعَى حَولَ الْحِمَى يُوشِكُ أَنْ يَرْتَعَ فِيهِ. أَلَا وَإِنَّ لِكُلِّ مَلِكٍ حِمًى، أَلَا وَإِنَّ حِمَى اللَّهِ مَحَارِمُهُ. أَلَا وَإِنَّ فِي الْجَسَدِ مُضْغَةً، إِذَا صَلَحَتْ صَلَحَ الْجَسَدُ كُلُّهُ، وَإِذَا فَسَدَتْ فَسَدَ الْجَسَدُ كُلُّهُ؛ أَلَا وَهِيَ الْقَلْبُ»

އަބޫ ޢަބްދު ﷲ، އައްނުޢްމާނު ބިން ބަޝީރު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”ހަމަކަށަވަރުން ޙަލާލު ކަންތައް ވަނީ ސާފުވެ ބަޔާން ވެގެންނެވެ. އަދި ހަމަކަށަވަރުން ޙަރާމް ކަންތައް ވެސް ވަނީ ސާފުވެ ބަޔާން ވެގެންނެވެ. އަދި އެ ދެމެދުގައި (ޙަރާމް ޙަލާލާ މެދު ޝައްކު އުފެދޭ) ޝުބުހާތަކެއް ވެއެވެ. މީސްތަކުންގެ ތެރެއިން ގިނަ ބަޔަކަށް އެ (ޙަލާލު ކަމެއް ނުވަތަ ޙަރާމް ކަމެއް) ނޭނގޭނެއެވެ. ފަހެ އެ ޝުބުހާތަކުން ރައްކާތެރިވެގެންފި މީހާ، އޭނާގެ ދީނާއި އަބުރަށް ޓަކައި (އެ ކަންތަކުން) ބަރީއަ ވެއްޖެއެވެ. އަދި އެ ޝުބުހާތަކަށް އަރައިގެންފި މީހާ، އޭނާ ޙަރާމް ކަންތަކަށް އަރައިގެންފިއެވެ. (އޭގެ މިސާލަކީ) ޙިމާޔަތް ކުރެވިފައިވާ އިމެއްގެ ވަށައިގެން ބަކަރި ހުއިހައްޕާ މީހެއްގެ މިސާލެވެ. ފަހެ، އެ ބަކަރިތައް އެ އިމަށް ވަދެ ހުއިކެއުމަކީ ގާތުގައިވާ ކަމެކެވެ. ދަންނާށެވެ! ހަމަކަށަވަރުން ކޮންމެ ރަސްކަލަކަށް ވެސް ހިމާޔަތް ކުރެވިފައިވާ އިމެއް ވެއެވެ. އަދި ދަންނާށެވެ! ހަމަކަށަވަރުން ﷲގެ އިމަކީ، އެ އިލާހު ޙަރާމް ކުރެއްވި ކަންތަކެވެ. އަދި ދަންނާށެވެ! ހަމަކަށަވަރުން (އިންސާނާގެ) ހަށިގަނޑުގައި މަސްކޮޅެއް ވެއެވެ. އެ މަސްކޮޅު އިސްލާހު ވެއްޖެ ނަމަ، މުޅި ހަށިގަނޑު އިސްލާހު ވެދެއެވެ. އަދި އެ މަސްކޮޅު ފާސިދު ވެއްޖެ ނަމަ، މުޅި ހަށިގަނޑު ފާސިދު ވެދެއެވެ. ދަންނާށެވެ! އެއީ ހިތެވެ.“

#07

[الدِّينُ النَّصِيحَةُ]

[ދީނަކީ ނަސޭހަތެވެ]

عَنْ أَبِي رُقَيَّةَ، تَمِيمِ بْنِ أَوسٍ الدَّارِيِّ رَضِيَ اللَّهُ عَنْهُ أَنَّ النَّبِيَّ ﷺ قَالَ: «الدِّينُ النَّصِيحَةُ» قُلْنَا: لِمَنْ؟ قَالَ: «لِلَّهِ، وَلِكِتَابِهِ، وَلِرَسُولِهِ، وَلِأَئِمَّةِ الْمُسْلِمِينَ، وَعَامَّتِهِمْ»

އަބޫ ރުޤައްޔާ، ތަމީމު ބިން އައުސް އައްދާރީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ނަބިއްޔާ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ދީނަކީ ނަސޭހަތެވެ.“ (އެބަހީ: އިޚްލާޞްތެރި ވުމެވެ.) ތިމަންމެން (ޞަޙާބީން) ދެންނެވީމެވެ: ކޮން ބަޔަކަށްތޯއެވެ؟ ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ އަށާއި، އެ އިލާހުގެ ފޮތަށާއި، އެ އިލާހުގެ ރަސޫލާއަށާ، އަދި މުސްލިމުންގެ ވެރިންނަށާއި، އެބައިމީހުންގެ އާންމުންނަށެވެ.“

#08

[أُمِرْتُ أَنْ أُقَاتِلَ النَّاسَ]

[ތިމަން ނަބިއްޔާއަށް ވަނީ މީސްތަކުންނާ ހަނގުރާމަ ކުރުމަށް އަމުރުވެވިގެންނެވެ]

عَنِ ابْنِ عُمَرَ رَضِيَ اللَّهُ عَنْهُمَا أَنَّ رَسُولَ اللَّهِ ﷺ قَالَ: «أُمِرْتُ أَنْ أُقَاتِلَ النَّاسَ حَتَّى يَشْهَدُوا أَلَّا إِلَهَ إِلَّا اللَّهُ وَأَنَّ مُحَمَّدًا رَسُولُ اللَّهِ، وَيُقِيمُوا الصَّلَاةَ، وَيُؤْتُوا الزَّكَاةَ. فَإِذَا فَعَلُوا ذَلِكَ عَصَمُوا مِنِّي دِمَاءَهُمْ وَأَمْوَالَهُمْ، إِلَّا بِحَقِّ الْإِسْلَامِ، وَحِسَابُهُمْ عَلَى اللَّهِ تَعَالَى»

އިބްނު ޢުމަރު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ ފިޔަވައި (އަޅުކަން ޙައްޤުވާ) އެހެން އެއްވެސް އިލާހަކު ނުވާކަމަށާއި، މުޙައްމަދުގެފާނީ ﷲގެ ރަސޫލާކަމަށް ހެކިވެ، އަދި ނަމާދު ގާއިމުކޮށް، (މުދަލު) ޒަކާތް ދީފުމަށް ދާންދެން، މީސްތަކުންނާ ހަނގުރާމަ ކުރުމަށް ތިމަން ނަބިއްޔާއަށް ވަނީ އަމުރުވެވިގެންނެވެ. ފަހެ އެބައިމީހުން އެ ފަދައިން ޢަމަލުކޮށްފި ނަމަ، އެބައިމީހުންގެ ލެޔާއި މުދާ ތިމަން ނަބިއްޔާގެ ކިބައިން ރައްކާތެރިކޮށް ސަލާމަތްކޮށްފިއެވެ. އެއީ އިސްލާމް ދީނުގެ ޙައްޤުގައި (އެބަހީ: ޝަރީޢަތާ އެއްގޮތަށް ގާއިމު ކުރެވޭ ހައްދެއްގައި) މެނުވީއެވެ. އަދި އެބައިމީހުންގެ ހިސާބު ބެއްލެވުން ވަނީ ﷲ ތަޢާލާއަށެވެ.“

#09

[مَا نَهَيتُكُمْ عَنْهُ فَاجْتَنِبُوهُ]

[ތިމަން ނަބިއްޔާ ނަހީ ކުރެއްވިހައި ކަމަކުން ތިޔަބައިމީހުން ދުރުވެގަންނާށެވެ]

عَنْ أَبِي هُرَيرَةَ، عَبْدِ الرَّحْمَنِ بْنِ صَخْرٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «مَا نَهَيتُكُمْ عَنْهُ فَاجْتَنِبُوهُ، وَمَا أَمَرْتُكُمْ بِهِ فَافْعَلُوا مِنْهُ مَا اسْتَطَعْتُمْ؛ فَإِنَّمَا أَهْلَكَ الَّذِينَ مِنْ قَبْلِكُمْ كَثْرَةُ مَسَائِلِهِمْ، وَاخْتِلَافُهُمْ عَلَى أَنْبِيَائِهِمْ»

އަބޫ ހުރައިރާ، ޢަބްދުއްރަޙްމާނު ބިން ޞަޚްރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”ތިމަން ނަބިއްޔާ ތިޔަބައިމީހުންނަށް ނަހީ ކުރެއްވިހައި ކަމަކުން ތިޔަބައިމީހުން ދުރުވެގަންނާށެވެ. އަދި ތިމަން ނަބިއްޔާ އަމުރު ކުރެއްވި ކަންތައްތައް، ތިޔަބައިމީހުންނަށް ކުޅަދާނަވި ވަރަކުން ކުރާށެވެ. ފަހެ ތިޔަބައިމީހުންގެ ކުރީގެ (އުންމަތްތަކުގެ) މީހުން ހަލާކު ކުރުވި ހަމައެކަނި ކަމަކީ، އެބައިމީހުން ސުވާލު ގިނަ ކުރުމާއި، އެބައިމީހުންގެ ނަބިއްޔުންނާ އެބައިމީހުން ޚިލާފު ވެގަތުމެވެ.“

#10

[إِنَّ اللَّهَ طَيِّبٌ لَا يَقْبَلُ إِلَّا طَيِّبًا]

[ﷲ ހުސްޠާހިރުވެވޮޑިގެންވެ، ޙަލާލު ރަނގަޅު އެއްޗެއް މެނުވީ ގަބޫލު ނުކުރައްވައެވެ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «إِنَّ اللَّهَ طَيِّبٌ لَا يَقْبَلُ إِلَّا طَيِّبًا، وَإِنَّ اللَّهَ تَعَالَى أَمَرَ الْمُؤْمِنِينَ بِمَا أَمَرَ بِهِ الْمُرْسَلِينَ، فَقَالَ تَعَالَى: ﴿يَا أَيُّهَا الرُّسُلُ كُلُوا مِنَ الطَّيِّبَاتِ وَاعْمَلُوا صَالِحًا﴾ وَقَالَ تَعَالَى: ﴿يَا أَيُّهَا الَّذِينَ آمَنُوا كُلُوا مِنْ طَيِّبَاتِ مَا رَزَقْنَاكُمْ﴾ ثُمَّ ذَكَرَ الرَّجُلَ، يُطِيلُ السَّفَرَ، أَشْعَثَ، أَغْبَرَ، يَمُدُّ يَدَيهِ إِلَى السَّمَاءِ: يَا رَبِّ! يَا رَبِّ! وَمَطْعَمُهُ حَرَامٌ، وَمَشْرَبُهُ حَرَامٌ، وَمَلْبَسُهُ حَرَامٌ، وَغُذِيَ بِالْحَرَامِ، فَأَنَّى يُسْتَجَابُ لِذَلِكَ؟!»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ﷲ އީ އެންމެ ރަނގަޅު ފަރާތެވެ. (އެބަހީ: އެންމެހައި އުނި ސިފަތަކުން ހުސްޠާހިރުވެ ވޮޑިގެންވާ އިލާހެވެ.) (ޙަލާލު) ރަނގަޅު އެއްޗެއް މެނުވީ އެ އިލާހު ގަބޫލު ނުކުރައްވައެވެ. އަދި ހަމަކަށަވަރުން ﷲ ތަޢާލާ މުއުމިނުންނަށް އަމުރު ކުރައްވަނީ، އެ އިލާހު ރަސޫލުންނަށް ވެސް އަމުރު ކުރެއްވި ކަންތައްތަކުންނެވެ. ފަހެ ﷲ ތަޢާލާ ވަޙީ ކުރެއްވިއެވެ: ❞އޭ ރަސޫލުންނޭވެ. (ޙަލާލު) ރަނގަޅު ތަކެތިން ތިޔަ ބޭކަލުން ފަރިއްކުޅުވައި އުޅުއްވާށެވެ. އަދި ޞާލިޙު ޢަމަލު ކުރައްވާށެވެ.❝ އަދިވެސް ﷲ ތަޢާލާ ވަޙީ ކުރެއްވިއެވެ: ❞އޭ އީމާންވި މީސްތަކުންނޭވެ. ތިމަން އިލާހު ދެއްވި (ޙަލާލު) ރަނގަޅު ތަކެތިން ތިޔަބައިމީހުން ކައި އުޅޭށެވެ.❝ ދެން އެއަށްފަހު، ދަތުރު ދިގުވެ، ބޯ ހިރަފުސްވެ، ގައި ކިލާވެފައިވާ މީހެއްގެ ވާހަކަ ނަބިއްޔާ ޒިކުރު ކުރެއްވިއެވެ. އެ މީހާ، ދެ އަތް އުޑާ ދިމާލަށް އުފުލައި (ދުޢާ ކުރެއެވެ:) ’އޭ އަޅުގެ ވެރި ރައްބެވެ! އޭ އަޅުގެ ވެރި ރައްބެވެ!‘ އެހެނަސް، އޭނާގެ ކާތަކެތި ޙަރާމެވެ. އަދި ބޯތަކެތި ވެސް ޙަރާމެވެ. އަދި ލާ ފޭރާން ވެސް ޙަރާމެވެ. އަދި ހެދިބޮޑުވެފައި ވަނީ ވެސް ޙަރާމް ތަކެތި ކައިގެންނެވެ. ފަހެ، އެފަދަ މީހެއްގެ ދުޢާ އިޖާބަ ވާނީ ކިހިނަކުންހެއްޔެވެ؟“

#11

[دَعْ مَا يُرِيبُكَ إِلَى مَا لَا يَرِيبُكَ]

[ތިބާއަށް ޝައްކުވާ ކަންތައް ދޫކޮށް، ޝައްކު ނުވާ ކަންތަކުގައި ހިފާށެވެ]

عَنْ أَبِي مُحَمَّدٍ، الْحَسَنِ بْنِ عَلِيِّ بْنِ أَبِي طَالِبٍ -سِبْطِ رَسُولِ اللَّهِ ﷺ، وَرَيحَانَتِهِ- رَضِيَ اللَّهُ عَنْهُمَا قَالَ: حَفِظْتُ مِنْ رَسُولِ اللَّهِ ﷺ: «دَعْ مَا يُرِيبُكَ إِلَى مَا لَا يَرِيبُكَ»

ރަސޫލު ﷲ ﷺ ގެ ލޯބިފުޅުވާ މަލެއް ފަދަ ކާފަދަރިކަލުން، އަބޫ މުޙައްމަދު އަލްޙަސަން ބިން ޢަލީ ބިން އަބީ ޠާލިބު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ އަރިހުން (މި ބަސްފުޅު) ތިމަން ދަސް ކުރީމެވެ: ”ތިބާއަށް ޝައްކުވާ ކަންތައް ދޫކޮށްލާފައި، ޝައްކު ނުވާ ކަންތަކުގައި ހިފާށެވެ.“

[مِنْ حُسْنِ إِسْلَامِ الْمَرْءِ]

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”މީހާގެ އިސްލާމްކަން ރަނގަޅުވެ (ފުރިހަމަ) ވުމުގެ ތެރޭގައި، އޭނާއާ ނުބެހޭ ކަންތައްތަކުގެ ތެރެއަށް ނުވަނުން ހިމެނެއެވެ.“

#13

[لَا يُؤْمِنُ أَحَدُكُمْ حَتَّى يُحِبَّ لِأَخِيهِ]

[އަމިއްލަ ނަފްސަށް ލޯބި ކުރާ ކަންތައް، ތިމާގެ އަޚާއަށް ވެސް ލޯބިކުރުން]

عَنْ أَبِي حَمْزَةَ، أَنَسِ بْنِ مَالِكٍ رَضِيَ اللَّهُ عَنْهُ -خَادِمِ رَسُولِ اللَّهِ ﷺ- عَنْ النَّبِيِّ ﷺ قَالَ: «لَا يُؤْمِنُ أَحَدُكُمْ حَتَّى يُحِبَّ لِأَخِيهِ مَا يُحِبُّ لِنَفْسِهِ»

ރަސޫލު ﷲ ﷺ ގެ ޚާދިމް އަބޫ ޙަމްޒާ، އަނަސް ބިން މާލިކު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ނަބިއްޔާ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާގެ އަމިއްލަ ނަފްސަށް ލޯބި ކުރާ ކަންތައް ތިބާގެ އަޚާއަށް ވެސް ލިބުމަށް ލޯބިކޮށްފުމަށް ދާންދެން، ތިޔަބައިމީހުންގެ ތެރެއިން އެކަކުވެސް (ޙަޤީޤީ ފުރިހަމަ ގޮތުގައި) އީމާންވީ ކަމުގައެއް ނުވާނެއެވެ.“

#14

[لَا يَحِلُّ دَمُ امْرِئٍ مُسْلِمٍ إِلَّا بِإِحْدَى ثَلَاثٍ]

[ތިން ކަމަކުން މެނުވީ މުސްލިމެއްގެ ލޭ ހުއްދަވެގެނެއް ނުވެއެވެ]

އިބްނު މަސްޢޫދު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިން ކަމަކުން މެނުވީ މުސްލިމެއްގެ ލޭ ހުއްދަ ވެގެނެއް ނުވެއެވެ. އެއީ ކާވެންޏަކަށް ފަހު ޒިނޭ ކުރުމާއި، (ޙައްޤަކާ ނުލައި ގަސްދުގައި މީހަކު މެރުމުން) ނަފްސަކަށް ނަފްސެއްގެ ބަދަލު ހިފުމާ، އަދި އެ މީހެއްގެ ދީން ދޫކޮށް (މުސްލިމުންގެ) ޖަމާޢަތާ ވަކިވެގަތުމެވެ.“

#15

[فَلْيَقُلْ خَيرًا أَو لِيَصْمُتْ]

[ހެޔޮ ބަސް ބުނާށެވެ. ނޫނީ ހަނު ހުންނާށެވެ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ أَنَّ رَسُولَ اللَّهِ ﷺ قَالَ: «مَنْ كَانَ يُؤْمِنُ بِاللَّهِ وَالْيَومِ الْآخِرِ؛ فَلْيَقُلْ خَيرًا أَو لِيَصْمُتْ. وَمَنْ كَانَ يُؤْمِنُ بِاللَّهِ وَالْيَومِ الْآخِرِ؛ فَلْيُكْرِمْ جَارَهُ. وَمَنْ كَانَ يُؤْمِنُ بِاللَّهِ وَالْيَومِ الْآخِرِ؛ فَلْيُكْرِمْ ضَيفَهُ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ އަށާއި އާޚިރަތް ދުވަހަށް އީމާންވާ މީހާ، ފަހެ އޭނާ ހެޔޮ ބަސް ބުނާށެވެ. ނޫންނަމަ ހަނު ހުންނާށެވެ. އަދި ﷲ އަށާއި އާޚިރަތް ދުވަހަށް އީމާންވާ މީހާ، ފަހެ އޭނާގެ އަވައްޓެރިޔާއަށް ހެޔޮކޮށް ހިތާށެވެ. އަދި ﷲ އަށާއި އާޚިރަތް ދުވަހަށް އީމާންވާ މީހާ، ފަހެ އޭނާގެ މެހުމާނަށް ހެޔޮކޮށް ހިތާށެވެ.“

[لَا تَغْضَبْ]

[ރުޅި އިސް ނުކުރާށެވެ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ: أَنَّ رَجُلًا قَالَ لِلنَّبِيِّ ﷺ: أَوصِنِي. قَالَ: «لَا تَغْضَبْ» فَرَدَّدَ مِرَارًا. قَالَ: «لَا تَغْضَبْ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ނަބިއްޔާ ﷺ ގެ އަރިހުގައި މީހަކު ދެންނެވިއެވެ: ތިމަންނާއަށް ވަޞިއްޔަތެއް ކޮށްދެއްވާށެވެ. (އެބަހީ: ނަސޭހަތެއް ދެއްވާށެވެ.) ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ރުޅި އިސް ނުކުރާށެވެ.“ ފަހެ، އޭނާ ތަކުރާރުކޮށް (އެ ފަދައިން ސުވާލު) ދެންނެވުމުން ވެސް ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ރުޅި އިސް ނުކުރާށެވެ.“

#17

[إِنَّ اللَّهَ كَتَبَ الْإِحْسَانَ عَلَى كُلِّ شَيءٍ]

[ﷲ ވަނީ ކޮންމެ ކަމެއްގައި ވެސް އިޙުސާން ކަނޑައަޅުއްވާފައެވެ]

عَنْ أَبِي يَعْلَى، شَدَّادِ بْنِ أَوسٍ رَضِيَ اللَّهُ عَنْهُ عَنْ رَسُولِ اللَّهِ ﷺ قَالَ: «إِنَّ اللَّهَ كَتَبَ الْإِحْسَانَ عَلَى كُلِّ شَيءٍ، فَإِذَا قَتَلْتُمْ فَأَحْسِنُوا الْقِتْلَةَ، وَإِذَا ذَبَحْتُمْ فَأَحْسِنُوا الذِّبْحَةَ، وَلْيُحِدَّ أَحَدُكُمْ شَفْرَتَهُ، وَلْيُرِحْ ذَبِيحَتَهُ»

އަބޫ ޔަޢުލާ، ޝައްދާދު ބިން އައުސް رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ﷲ ވަނީ ކޮންމެ ކަމެއްގައި ވެސް އިޙުސާން ކަނޑައަޅުއްވާ ލިޔުއްވާފައެވެ. (އެބަހީ: ކުރެވެން ހުރި އެންމެ ރަނގަޅަކަށް ކަންތައް ކުރުން، ނުވަތަ އިޙުސާންތެރިވުން ވާޖިބު ކުރައްވާފައެވެ.) ވީމާ ތިޔަބައިމީހުން (ޤަތުލު ކުރަން ހުއްދަ އެއްޗެއް) ޤަތުލު ކުރާ ނަމަ، ފަހެ އެންމެ ރަނގަޅު ގޮތުގައި ޤަތުލު ކުރާށެވެ. އަދި ތިޔަބައިމީހުން (ޖަނަވާރެއް) ކަތިލާ ނަމަ، ފަހެ އެންމެ ރަނގަޅު ގޮތުގައި ކަތިލާށެވެ. އަދި ތިޔަބައިމީހުންގެ ތެރެއިން ކޮންމެ މީހަކުމެ އޭނާގެ ވަޅި ތޫނުކޮށް، އެ މީހެއްގެ ޖަނަވާރަށް ވީހާވެސް ލުއިވާ ގޮތަކަށް ކަތިލާށެވެ.“

#18

[اتَّقِ اللَّهَ حَيثُمَا كُنْتَ]

[ތިބާވީ ކޮންމެ ތާކުމެ ﷲ އަށް ތަޤުވާވެރި ވާށެވެ]

عَنْ أَبِي ذِرٍّ جُنْدُبِ بْنِ جُنَادِةَ، وَأَبِي عَبْدِ الرَّحْمَنِ مُعَاذِ بْنِ جَبَلٍ رَضِيَ اللَّهُ عَنْهُمَا عَنْ رَسُولِ اللَّهِ ﷺ قَالَ: «اتَّقِ اللَّهَ حَيثُمَا كُنْتَ، وَأَتْبِعِ السَّيِّئَةَ الْحَسَنَةَ تَمْحُهَا، وَخَالِقِ النَّاسَ بِخُلُقٍ حَسَنٍ»

އަބޫ ޛައްރު ޖުންދުބް ބިން ޖުނާދާއާއި އަބޫ އަބްދުއްރަޙްމާނު މުޢާޛު ބިން ޖަބަލު رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާވީ ކޮންމެ ތާކުމެ ﷲ އަށް ތަޤުވާވެރި ވާށެވެ. އަދި (ފާފައެއް ކުރެވިއްޖެ ނަމަ، އެ) ފާފައަކާ ވިދިގެން ހެޔޮ ކަމެއް ކުރާށެވެ. (އޭރުން) އެ ހެޔޮ ކަންތައް، އެ ފާފަ ފޮހެލާނެއެވެ. އަދި ރިވެތި އަޚްލާޤުން މީސްތަކުންނާ މުޢާމަލާތް ކުރާށެވެ.“

#19

[احْفَظِ اللَّهَ يَحْفَظْكَ]

[ﷲ އާމެދު ރައްކާތެރިވޭ، އޭރުން އެ އިލާހު ތިބާ ރައްކާތެރި ކޮށްދެއްވާނެއެވެ]

عَنْ أَبِي الْعَبَّاسِ، عَبْدِ اللَّهِ بْنِ عَبَّاسٍ رَضِيَ اللَّهُ عَنْهُمَا قَالَ: كُنْتُ خَلْفَ النَّبِيِّ ﷺ يَومًا فَقَالَ: «يَا غُلَامُ! إِنِّي أُعَلِّمُكَ كَلِمَاتٍ: احْفَظِ اللَّهَ يَحْفَظْكَ، احْفَظِ اللَّهَ تَجِدْهُ تُجَاهَكَ. إِذَا سَأَلْتَ فَاسْأَلِ اللَّهَ، وَإِذَا اسْتَعَنْتَ فَاسْتَعِنْ بِاللَّهِ. وَاعْلَمْ: أَنَّ الْأُمَّةَ لَوِ اجْتَمَعَتْ عَلَى أَنْ يَنْفَعُوكَ بِشَيءٍ، لَمْ يَنْفَعُوكَ إِلَّا بِشَيءٍ قَدْ كَتَبَهُ اللَّهُ لَكَ. وَإِنْ اجْتَمَعُوا عَلَى أَنْ يَضُرُّوكَ بِشَيءٍ، لَمْ يَضُرُّوكَ إِلَّا بِشَيءٍ قَدْ كَتَبَهُ اللَّهُ عَلَيكَ، رُفِعَتِ الْأَقْلَامُ، وَجَفَّتِ الصُّحُفُ» رَوَاهُ التِّرْمِذِيُّ، وَقَالَ: ”حَدِيثٌ حَسَنٌ صَحِيحٌ“.
وَفِي رِوَايَةِ غَيرِ التِّرْمِذِيِّ: «احْفَظِ اللَّهَ تَجِدْهُ أَمَامَكَ، تَعَرَّفْ إِلَى اللَّهِ فِي الرَّخَاءِ يَعْرِفُكَ فِي الشِّدَّةِ. وَاعْلَمْ: أَنَّ مَا أَخْطَأَكَ لَمْ يَكُنْ لِيُصِيبَكَ، وَمَا أَصَابَكَ لَمْ يَكُنْ لِيُخْطِئَكَ. وَاعْلَمْ: أَنَّ النَّصْرَ مَعَ الصَّبْرِ، وَأَنَّ الْفَرَجَ مَعَ الْكَرْبِ، وَأَنَّ مَعَ الْعُسْرِ يُسْرًا»

އަބުލް ޢައްބާސް، ޢަބްދު ﷲ ބިން ޢައްބާސް رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: އެއް ދުވަހަކު ނަބިއްޔާ ﷺ ގެ ފަހަތްޕުޅުގައި ތިމަން ވަނިކޮށް، ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އޭ ކޮއިފުޅާއެވެ. ހަމަކަށަވަރުން ތިމަން ނަބިއްޔާ ތިބާއަށް ކަލިމަތަކެއް ދަސްކޮށް ދެއްވާނަމެވެ. ﷲ އާމެދު (އެބަހީ: އެ އިލާހުގެ އަމުރުފުޅުތަކާ މެދު) ތިބާ ރައްކާތެރި ވާށެވެ. އޭރުން އެ އިލާހު ތިބާ ރައްކާތެރި ކޮށްދެއްވާނެއެވެ. ﷲ އާމެދު ތިބާ ރައްކާތެރި ވާށެވެ. އޭރުން ތިބާގެ ކުރިމަތީގައި އެ އިލާހު ވޮޑިގެންވާނެއެވެ. (އެބަހީ: ތިބާ ﷲ އަށް ކުއްތަންވެ، އެ އިލާހުގެ އަމުރުފުޅުތަކަށް ރައްކާތެރި ވާނަމަ، ﷲ ވެސް ތިބާއާ ކުއްތަންވެ ވޮޑިގަންނަވާ، ތިބާ ރައްކާތެރިކޮށް، ތިބާގެ ދުޢާތައް އިޖާބަ ކުރައްވާނެއެވެ.) ތިބާ ކަމަކަށް އެދި ދަންނަވާ ނަމަ، ﷲ އަށް އެދި ދަންނަވާށެވެ. އަދި ތިބާ އެހީއަކަށް އެދޭ ނަމަ، ﷲގެ އެހީއަށް އެދޭށެވެ. އަދި ދަންނާށެވެ. ހަމަކަށަވަރުން މުޅި އުންމަތް، ތިބާއަށް އެއްވެސް މަންފާއެއް ކޮށްދިނުމަށް އެއްބައި ވިޔަސް، ﷲ ތިބާއަށް ޓަކައި ލިޔުއްވައި (މިންވަރު ކުރައްވައިފި) ކަމަކުން މެނުވީ އެބައިމީހުން ތިބާއަކަށް މަންފާއެއް ކޮށެއް ނުދެވޭނެއެވެ. އަދި ތިބާއަށް އެއްވެސް ގެއްލުމެއް ދިނުމަށް މުޅި އުންމަތް އެއްބައި ވިޔަސް، ﷲ ތިބާގެ މައްޗަށް ލިޔުއްވައި (މިންވަރު ކުރައްވައިފި) ކަމަކުން މެނުވީ އެބައިމީހުން ތިބާއަށް ގެއްލުމެއް ނުދެވޭނެއެވެ. ގަލަންތައް ވަނީ އުފުލިފައެވެ. އަދި ފަތްފުއްތައް ވަނީ ހިކިފައެވެ.“ (އެބަހީ: ލައުޙުލް މަޙްފޫޡުގައި ﷲ ކަނޑައެޅުއްވި ކަންތައް ލިޔެ ނިމި، އެ ގޮތަށް އެ ކަންތައް ހިނގާނެއެވެ.)
އަދި ތިރްމިޛީ ނޫން ރިވާޔަތެއްގައި ވެއެވެ: ”ﷲ އާމެދު ތިބާ ރައްކާތެރި ވާށެވެ. އޭރުން ތިބާގެ ކުރިމަތީގައި އެ އިލާހު ވޮޑިގެންވާނެއެވެ. އަދި ފަސޭހަކަމާއި ތިޔާގިކަމުގެ ޙާލުގައި ﷲ ދެނެގަންނާށެވެ. (އެބަހީ: އެ ޙާލުގައި ﷲ ހަނދުމަކޮށް، އެ އިލާހު އެންގެވި ގޮތަށް އުޅޭށެވެ.) އޭރުން ތިބާގެ ދަތިކަމާއި އުނދަގޫކަމުގެ ޙާލުގައި، އެ އިލާހު ތިބާ ދެނެވޮޑިގެންވާނެއެވެ. (އެބަހީ: އެ ޙާލުގައި ތިބާގެ ކަންކަން ފަސޭހަ ކޮށްދެއްވާނެއެވެ.) އަދި ދަންނާށެވެ! ތިބާއަށް ނުޖެހި ފަހަނަޅައި ހިނގައްޖެ ކަމެއް، އެއީ ތިބާއަށް ޖެހެން އޮތް ކަމެއް ކަމުގައި ނުވެއެވެ. އަދި ތިބާއަށް ޖެހިއްޖެ ކަމެއް، އެއީ ތިބާއަށް ނުޖެހި ފަހަނަޅައި ދާނޭ ކަމެއްކަމުގައި ވެސް ނުވެއެވެ. އަދި ދަންނާށެވެ! ހަމަކަށަވަރުން ކެތްތެރިކަމާއެކު ނަޞްރު ވެއެވެ. އަދި ދަތިކަމާއެކު ޙައްލު ވެއެވެ. އަދި އުނދަގޫކަމާއެކު ފަސޭހަކަން ވެއެވެ.“

[إِذَا لَمْ تَسْتَحْيِ فَاصْنَعْ مَا شِئْتَ]

[ތިބާ ލަދު ނުގަންނަ ނަމަ ހިތަށް އެރިހައި ގޮތަކަށް އުޅޭށެވެ]

عَنْ أَبِي مَسْعُودٍ، عُقْبَةَ بْنِ عَمْرٍو الْأَنْصَارِيِّ الْبَدْرِيِّ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «إِنَّ مِمَّا أَدْرَكَ النَّاسُ مِنْ كَلَامِ النُّبُوَّةِ الْأُولَى: إِذَا لَمْ تَسْتَحْيِ فَاصْنَعْ مَا شِئْتَ»

އަބޫ މަސްޢޫދު، ޢުޤްބާ ބިން ޢަމްރު އަލްއަންޞާރީ އަލްބަދްރީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން، އިހު ޒަމާނުގެ (އުންމަތްތަކުގެ) ނަބިއްޔުންގެ ފަރާތުން މީސްތަކުންނާ ހަމައަށް ފޯރާފައިވާ ބަސްފުޅުތަކުގެ ތެރޭގައި ވެއެވެ: ތިބާގެ ލަދެއް (ހަޔާތެއް) ނެތް ނަމަ، (ނިކަން) ހިތަށް އެރިހައި ގޮތަކަށް އުޅެބަލާށެވެ.“ (ނުވަތަ މާނައަކީ: ލަދުގަންނަން ޖެހޭ (ފާފަވެރި) ކަމަކަށް ނުވާނަމަ، ތިބާ ކުރަން ބޭނުންވާ ކަމެއް ކުރާށެވެ.)

[قُلْ آمَنْتُ بِاللَّهِ ثُمَّ اسْتَقِمْ]

[ﷲ އަށް އީމާން ވެއްޖައީމުއޭ ބުނެ، އެއަށްފަހު ސާބިތުވެ ހުންނާށެވެ]

عَنْ أَبِي عَمْرٍو -وَقِيلَ أَبِي عَمْرَةَ-، سُفْيَانَ بْنِ عَبْدِ اللَّهِ رَضِيَ اللَّهُ عَنْهُ قَالَ: قُلْتُ: يَا رَسُولَ اللَّهِ! قُلْ لِي فِي الْإِسْلَامِ قَولًا لَا أَسْأَلُ عَنْهُ أَحَدًا غَيرَكَ. قَالَ: «قُلْ آمَنْتُ بِاللَّهِ، ثُمَّ اسْتَقِمْ»

އަބޫ ޢަމްރު -އަދި އަބޫ ޢަމްރާ ކަމަށް ވެސް ބުނެވުނެވެ- ސުފްޔާނު ބިން ޢަބްދު ﷲ އައްޘަޤަފީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ތިމަން ދެންނެވީމެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. އިސްލާމް ދީނާއި ގުޅޭ ގޮތުން، އިބަ ރަސޫލާއަށް ފަހު އެހެން އެއްވެސް ފަރާތަކާ އިތުރަށް ސުވާލު ކުރަން ނުޖެހޭ ފަދަ (ޝާމިލު) ބަސްފުޅެއް ތިމަންނާއަށް ބުނެދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ އަށް އީމާން ވެއްޖައީމުއޭ ބުނާށެވެ. އެއަށްފަހު (އެ އިލާހު އެންގެވި ގޮތުގެ މަތީ) ސާބިތުވެ ހުންނާށެވެ.“

[أَرَأَيتَ إِذَا صَلَّيتُ الْمَكْتُوبَاتِ]

[ފަރުޟު ފަސް ނަމާދު ކުރާ ނަމަ]

عَنْ أَبِي عَبْدِ اللَّهِ، جَابِرِ بْنِ عَبْدِ اللَّهِ الْأَنْصَارِيِّ رَضِيَ اللَّهُ عَنْهُمَا: أَنَّ رَجُلًا سَأَلَ رَسُولَ اللَّهِ ﷺ فَقَالَ: أَرَأَيتَ إِذَا صَلَّيتُ الْمَكْتُوبَاتِ، وَصُمْتُ رَمَضَانَ، وَأَحْلَلْتُ الْحَلَالَ، وَحَرَّمْتُ الْحَرَامَ، وَلَمْ أَزِدْ عَلَى ذَلِكَ شَيئًا؛ أَأَدْخُلُ الْجَنَّةَ؟ قَالَ: «نَعَمْ» رَوَاهُ مُسْلِمٌ.
وَمَعْنَى ”حَرَّمْتُ الْحَرَامَ“: اجْتَنَبْتُهُ.
وَمَعْنَى ”أَحْلَلْتُ الْحَلَالَ“: فَعَلْتُهُ مُعْتَقِدًا حِلَّهُ.

އަބޫ ޢަބްދު ﷲ، ޖާބިރު ބިން ޢަބްދު ﷲ އަލްއަންޞާރީ رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ އަރިހުގައި މީހަކު ސުވާލު ކުރެއްވިއެވެ: ތިމަން ފަރުޟު ފަސް ނަމާދުކޮށް، ރަމަޟާން މަހު ރޯދަ ހިފައި، ޙަލާލު ކަންތައް ޙަލާލު ކަމުގައި ބަލައިގަނެ (އެ ކަންތައް ކޮށް)، ޙަރާމް ކަންތައް ޙަރާމް ކަމުގައި ބަލައިގަނެ (އެ ކަންކަމުން ދުރުވެ)، އަދި އޭގެ އިތުރަށް އެއްވެސް (ހެޔޮ) ކަމެއް ނުކުރިކަމުގައި ވިޔަސް، ތިމަން ސުވަރުގެ ވަންނާނެކަމަށް އިބަ ނަބިއްޔާ ދެކިލައްވަމުތޯއެވެ؟ ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އާދޭހެވެ.“

#23

[الطَّهُورُ شَطْرُ الْإِيمَانِ]

[ސާފުތާހިރުކަމަކީ އީމާންކަމުގެ ދެބައިކުޅަ އެއްބައެވެ]

عَنْ أَبِي مَالِكٍ، الْحَارِثِ بْنِ عَاصِمٍ الْأَشْعَرِيِّ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «الطَّهُورُ شَطْرُ الْإِيمَانِ. وَالْحَمْدُ لِلَّهِ تَمْلَأُ الْمِيزَانَ. وَسُبْحَانَ اللَّهِ وَالْحَمْدُ لِلَّهِ تَمْلَآنِ -أَو تَمْلَأُ- مَا بَينَ السَّمَاءِ وَالْأَرْضِ. وَالصَّلَاةُ نُورٌ، وَالصَّدَقَةُ بُرْهَانٌ، وَالصَّبْرُ ضِيَاءٌ. وَالْقُرْآنُ حُجَّةٌ لَكَ أَو عَلَيكَ. كُلُّ النَّاسِ يَغْدُو، فَبَائِعٌ نَفْسَهُ، فَمُعْتِقُهَا، أَو مُوبِقُهَا»

އަބޫ މާލިކު، އަލްޙާރިޘް ބިން ޢާޞިމު އަލްއަޝްޢަރީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ސާފުތާހިރުކަމަކީ އީމާންކަމުގެ ދެބައިކުޅަ އެއްބައެވެ. އަދި الْحَمْدُ لِلَّهِ (މި ކަލިމަ، ޢަމަލުތަކުގެ) މީޒާން ފުރާލައެވެ. އަދި سُبْحَانَ اللَّهِ وَالْحَمْدُ لِلَّهِ (މި ދެ ކަލިމަ) އުޑާއި ބިމާ ދެމެދު ފުރާލައެވެ. އަދި ނަމާދަކީ ނޫރެކެވެ. އަދި ޞަދަޤާތަކީ (އީމާންކަމުގެ) ބުރުހާނެކެވެ. އަދި ކެތްތެރިކަމަކީ އަލިކަމެކެވެ. އަދި ޤުރުއާނަކީ ތިބާއާ އެއްކޮޅަށް ނުވަތަ ދެކޮޅަށްވާ ހުއްޖަތެކެވެ. ކޮންމެ މީހަކު ވެސް ހެނދުނު ހެނދުނާ ނުކުމެގެން ދަނީ އޭނާގެ ނަފްސުގެ ވިޔަފާރިވެރިއެއްގެ ގޮތުގައެވެ. އެ ގޮތުން (ހެޔޮ ޢަމަލުތައްކޮށް) ނަފްސު ސަލާމަތް ކުރާ ބަޔަކާ، (ފާފަތަކަށް އަރައިގަނެ) ނަފްސު ހަލާކު ކުރާ ބަޔަކުވެއެވެ.“

#24

[يَا عِبَادِي، إِنِّي حَرَّمْتُ الظُّلْمَ عَلَى نَفْسِي]

[ތިމަން އިލާހުގެ ނަފްސުފުޅުގެ މައްޗަށް އަނިޔާވެރިކަން ޙަރާމް ކުރެއްވީމެވެ]

عَنْ أَبِي ذَرٍّ رَضِيَ اللَّهُ عَنْهُ عَنِ النَّبِيِّ ﷺ فِيمَا رَوَى عَنِ اللَّهِ عَزَّ وَجَلَّ أَنَّهُ قَالَ: «يَا عِبَادِي! إِنِّي حَرَّمْتُ الظُّلْمَ عَلَى نَفْسِي، وَجَعَلْتُهُ بَينَكُمْ مُحَرَّمًا فَلَا تَظَالَمُوا. يَا عِبَادِي! كُلُّكُمْ ضَالٌّ إِلَّا مَنْ هَدَيتُهُ، فَاسْتَهْدُونِي أَهْدِكُمْ. يَا عِبَادِي! كُلُّكُمْ جَائِعٌ إِلَّا مَنْ أَطْعَمْتُهُ، فَاسْتَطْعِمُونِي أُطْعِمْكُمْ. يَا عِبَادِي! كُلُّكُمْ عَارٍ إِلَّا مَنْ كَسَوتُهُ، فَاسْتَكْسُونِي أَكْسُكُمْ. يَا عِبَادِي! إِنَّكُمْ تُخْطِئُونَ بِاللَّيلِ وَالنَّهَارِ، وَأَنَا أَغْفِرُ الذُّنُوبَ جَمِيعًا، فَاسْتَغْفِرُونِي أَغْفِرْ لَكُمْ. يَا عِبَادِي! إِنَّكُمْ لَنْ تَبْلُغُوا ضَرِّي فَتَضُرُّونِي، وَلَنْ تَبْلُغُوا نَفْعِي فَتَنْفَعُونِي. يَا عِبَادِي! لَو أَنَّ أَوَّلَكُمْ وَآخِرَكُمْ، وَإِنْسَكُمْ وَجِنَّكُمْ، كَانُوا عَلَى أَتْقَى قَلْبِ رَجُلٍ وَاحِدٍ مِنْكُمْ مَا زَادَ ذَلِكَ فِي مُلْكِي شَيئًا. يَا عِبَادِي! لَو أَنَّ أَوَّلَكُمْ وَآخِرَكُمْ، وَإِنْسَكُمْ وَجِنَّكُمْ، كَانُوا عَلَى أَفْجَرِ قَلْبِ رَجُلٍ وَاحِدٍ، مَا نَقَصَ ذَلِكَ مِنْ مُلْكِي شَيئًا. يَا عِبَادِي! لَو أَنَّ أَوَّلَكُمْ وَآخِرَكُمْ، وَإِنْسَكُمْ وَجِنَّكُمْ، قَامُوا فِي صَعِيدٍ وَاحِدٍ، فَسَأَلُونِي، فَأَعْطَيتُ كُلَّ إِنْسَانٍ مَسْأَلَتَهُ، مَا نَقَصَ ذَلِكَ مِمَّا عِنْدِي إِلَّا كَمَا يَنْقُصُ المِخْيَطُ إِذَا أُدْخِلَ الْبَحْرَ. يَا عِبَادِي! إِنَّمَا هِيَ أَعْمَالُكُمْ أُحْصِيهَا لَكُمْ، ثُمَّ أُوَفِّيكُمْ إِيَّاهَا، فَمَنْ وَجَدَ خَيرًا، فَلْيَحْمَدِ اللَّهَ، وَمَنْ وَجَدَ غَيرَ ذَلِكَ، فَلَا يَلُومَنَّ إِلَّا نَفْسَهُ»

އަބޫ ޛައްރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ﷲ ޢައްޒަ ވަޖައްލަގެ ޙަޟްރަތުން، ނަބިއްޔާ ﷺ ރިވާ ކުރެއްވި (ޤުދުސީ ޙަދީޘެއްގައި ވެއެވެ.) ﷲ ވަޙީ ކުރެއްވިއެވެ: ”އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ހަމަކަށަވަރުން ތިމަން އިލާހުގެ ނަފްސުފުޅުގެ މައްޗަށް އަނިޔާވެރިކަން ޙަރާމް ކުރެއްވީމެވެ. އަދި ތިޔަބައިމީހުންގެ މެދުގައި ވެސް އެއީ ޙަރާމް ކަމެއް ކަމުގައި ތިމަން އިލާހު ލެއްވީމެވެ. ވީމާ، ތިޔަބައިމީހުން އެއްބަޔަކު އަނެއް ބަޔަކަށް އަނިޔާވެރި ނުވާށެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިމަން އިލާހު ހިދާޔަތު ދައްކަވައިފި މީހަކު ފިޔަވައި ތިޔަބައިމީހުން އެންމެން ވެސް ވަނީ މަގުފުރެދިގެންނެވެ. ވީމާ، ތިމަން އިލާހުގެ ޙަޟްރަތުން ހިދާޔަތު ދެއްވުމަށް އެދި ދަންނަވާށެވެ. އޭރުން ތިމަން އިލާހު ތިޔަބައިމީހުންނަށް ހިދާޔަތް ދައްކަވާނަމެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިމަން އިލާހު ކާން ދެއްވައިފި މީހަކު ފިޔަވައި ތިޔަބައިމީހުން އެންމެން ވެސް ވަނީ ބަނޑުހައި ހޫނުކަންމަތީގައެވެ. ވީމާ، ތިމަން އިލާހުގެ ޙަޟްރަތުން ކާންދެއްވުމަށް އެދި ދަންނަވާށެވެ. އޭރުން ތިމަން އިލާހު ތިޔަބައިމީހުންނަށް ކާން ދެއްވާނަމެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިމަން އިލާހު ފޭރާން ދެއްވައިފި މީހަކު ފިޔަވައި ތިޔަބައިމީހުން އެންމެން ވަނީ އޮރިޔާން ޙާލުގައެވެ. ވީމާ، ތިމަން އިލާހުގެ ޙަޟްރަތުން ފޭރާން ދެއްވުމަށް އެދި ދަންނަވާށެވެ. އޭރުން ތިމަން އިލާހު ތިޔަބައިމީހުންނަށް ފޭރާން ދެއްވާނަމެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ހަމަކަށަވަރުން ތިޔަބައިމީހުން ރެއާއި ދުވާލު ކުށްފާފަތަކަށް އަރައިގަންނަމުއެވެ. އަދި ތިމަން އިލާހު އެންމެހައި ކުށްފާފަތަކެއް ފުއްސަވައި ދެއްވަމެވެ. ވީމާ، ތިމަން އިލާހުގެ ޙަޟްރަތުން ފާފަ ފުއްސަވައި ދެއްވުމަށް އެދި ދަންނަވާށެވެ. އޭރުން ތިމަން އިލާހު ތިޔަބައިމީހުންނަށް ފާފަ ފުއްސަވައި ދެއްވާނަމެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ހަމަކަށަވަރުން ތިމަން އިލާހަށް ގެއްލުމެއް ދޭން ތިޔަބައިމީހުން އުޅުނުކަމުގައި ވިޔަސް، ތިމަން އިލާހަށް އެއްވެސް ގެއްލުމެއް ނުދެވޭނެއެވެ. އަދި ތިމަން އިލާހަށް މަންފާއެއް ކޮށްދޭން ތިޔަބައިމީހުން އުޅުނުކަމުގައި ވިޔަސް، ތިމަން އިލާހަށް އެއްވެސް މަންފާއެއްކޮށެއް ނުދެވޭނެއެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިޔަބައިމީހުންގެ ފުރަތަމަ މީހާއާއި، ފަހު މީހާއާއި، އިންސީންނާއި، ޖިންނީންގެ އެންމެން، ތިޔަބައިމީހުންގެ ތެރެއިންވާ އެންމެ ތަޤުވާވެރި ހިތެއްގެ ވެރިޔާ ފަދައިންވީ ނަމަވެސް، އެ ކަމުގެ ސަބަބުން ތިމަން އިލާހުގެ މިލްކުވެރިކަމަށް އެއްވެސް އެއްޗެއް އިތުރުވެގެނެއް ނުދާނެއެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިޔަބައިމީހުންގެ ފުރަތަމަ މީހާއާއި، ފަހު މީހާއާއި، އިންސީންނާއި، ޖިންނީންގެ އެންމެން، އެންމެ ފާޖިރު ހިތެއްގެ ވެރިޔާ ފަދައިން ވީނަމަ ވެސް، އެ ކަމުގެ ސަބަބުން ތިމަން އިލާހުގެ މިލްކުވެރިކަމުން އެއްވެސް އެއްޗެއް އުނިވެގެނެއް ނުދާނެއެވެ. އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިޔަބައިމީހުންގެ ފުރަތަމަ މީހާއާއީ، ފަހު މީހާއާއި، އިންސީންނާއި، ޖިންނީންގެ އެންމެން، އެއް ތަނެއްގައި އެއްވެ ތިބެ، އެބައިމީހުން ތިމަން އިލާހުގެ ޙަޟްރަތުގައި އެދި ދެންނެވުމުން، އެއިން ކޮންމެ މީހަކަށް އެ މީހަކު އެދޭ ކަމެއް ދެއްވިކަމުގައި ވިޔަސް، އޭގެ ސަބަބުން ކަނޑަކަށް ތިނޯހެއް ލާފައި ނެގުމުން، އެ ކަނޑުން އުނިވާ ފަދައިން މެނުވީ ތިމަން އިލާހުގެ ޙަޟްރަތުގައިވާ އެއްވެސް އެއްޗަކުން އުނިވެގެން ނުދާނެއެވެ. (އެބަހީ: އެއްވެސް މިންވަރަކަށް މަދުވެގެނެއް ނުދާނެއެވެ.) އޭ ތިމަން އިލާހުގެ އަޅުތަކުންނޭވެ. ތިމަން އިލާހު ތިޔަބައިމީހުންނަށް ހިސާބު ބައްލަވަނީ ހަމައެކަނި ތިޔަބައިމީހުންގެ ޢަމަލުތަކުންނެވެ. ދެން އެއަށްފަހު، އެ ޢަމަލުތަކާ އެކަށޭނެ ޖަޒާ ތިޔަބައިމީހުންނަށް ދެއްވަމެވެ. ވީމާ ހެޔޮ ގޮތެއް ފެނިއްޖެ މީހާ، ފަހެ ﷲ އަށް ޙަމްދު ކުރާށެވެ. އަދި އެ ނޫން ގޮތެއް ފެނިއްޖެ މީހާ، ފަހެ އޭނާގެ އަމިއްލަ ނަފްސަށް މެނުވީ އެއްގޮތަކަށް ވެސް މަލާމާތް ނުކުރާށެވެ.“ (އެބަހީ: ކުށްވެރިކޮށް ނުހަދާށެވެ.)

#25

[إِنَّ بِكُلِّ تَسْبِيحَةٍ صَدَقَةً]

[ކޮންމެ ތަސްބީޙައެއްގައި ޞަދަޤާތެއް ވެއެވެ]

عَنْ أَبِي ذَرٍّ رَضِيَ اللَّهُ عَنْهُ -أَيضًا-: أَنَّ نَاسًا مِنْ أَصْحَابِ رَسُولِ اللَّهِ ﷺ قَالُوا لِلنَّبِيِّ ﷺ: يَا رَسُولَ اللَّهِ! ذَهَبَ أَهْلُ الدُّثُورِ بِالْأُجُورِ، يُصَلُّونَ كَمَا نُصَلِّي، وَيَصُومُونَ كَمَا نَصُومُ، وَيَتَصَدَّقُونَ بِفُضُولِ أَمْوَالِهِمْ. قَالَ: «أَوَلَيسَ قَدْ جَعَلَ اللَّهُ لَكُمْ مَا تَصَّدَّقُونَ؟! إِنَّ بِكُلِّ تَسْبِيحَةٍ صَدَقَةً، وَكُلِّ تَكْبِيرَةٍ صَدَقَةً، وَكُلِّ تَحْمِيدَةٍ صَدَقَةً، وَكُلِّ تَهْلِيلَةٍ صَدَقَةً. وَأَمْرٌ بِالْمَعْرُوفِ صَدَقَةٌ، وَنَهْيٌ عَنْ مُنْكَرٍ صَدَقَةٌ. وَفِي بُضْعِ أَحَدِكُمْ صَدَقَةٌ» قَالُوا: يَا رَسُولَ اللَّهِ! أَيَأْتِي أَحَدُنَا شَهْوَتَهُ، وَيَكُونُ لَهُ فِيهَا أَجْرٌ؟! قَالَ: «أَرَأَيتُمْ لَو وَضَعَهَا فِي حَرَامٍ أَكَانَ عَلَيهِ وِزْرٌ؟ فَكَذَلِكَ إِذَا وَضَعَهَا فِي الْحَلَالِ، كَانَ لَهُ أَجْرٌ»

އަދިވެސް އަބޫ ޛައްރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ ޞަޙާބީންގެ ތެރެއިން ބަޔަކު، އެ ނަބިއްޔާގެ އަރިހު ދެންނެވިއެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. މުދަލުގެ އަހުލުވެރީން (ހުރިހާ) އަޖުރަކައިގެން ހިނގައްޖެއެވެ. ތިމަންމެން ނަމާދު ކުރާ ފަދައިން، އެބައިމީހުން ވެސް ނަމާދު ކުރެއެވެ. އަދި ތިމަންމެން ރޯދަ ހިފާ ފަދައިން، އެބައިމީހުން ވެސް ރޯދަ ހިފައެވެ. އަދި އެބައިމީހުންގެ (އަތުގައި) އިތުރުވެފައިވާ މުދަލުން އެބައިމީހުން ޞަދަޤާތްދީ ހަދައެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިޔަބައިމީހުންނަށް ޞަދަޤާތް ކުރެވޭނެ ގޮތެއް ﷲ ނުލައްވާހެއްޔެވެ؟ ހަމަކަށަވަރުން ކޮންމެ ތަސްބީޙައެއްގައި ޞަދަޤާތެއް ވެއެވެ. އަދި ކޮންމެ ތަކުބީރެއްގައި ޞަދަޤާތެއް ވެއެވެ. އަދި ކޮންމެ ތަޙްމީދެއްގައި ޞަދަޤާތެއް ވެއެވެ. އަދި ކޮންމެ ތަހްލީލެއްގައި ޞަދަޤާތެއް ވެއެވެ. (އެބަހީ: سُبْحَانَ اللَّهِ، اللَّهُ أَكْبَرُ، اَلْحَمْدُ لِلَّهِ، لَا إِلٰهَ إِلَّا اللَّهُ މިއިން ކޮންމެ ޒިކުރެއް ކިޔުމުން ސަވާބު ލިބޭނެއެވެ.) އަދި ހެޔޮ ކަމަކަށް އަމުރު ކުރުމަކީ ޞަދަޤާތެކެވެ. އަދި ނުބައި ކަމެއް މަނާ ކުރުމަކީ ޞަދަޤާތެކެވެ. އަދި ތިޔަބައިމީހުންގެ ތެރެއިން މީހަކު، އޭނާގެ (އަނބިމީހާއާއެކު) ޖިމާޢުވުމުގައި ވެސް ޞަދަޤާތެއް ވެއެވެ.“ ޞަޙާބީން ދެންނެވިއެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. ތިމަންމެންގެ ތެރެއިން މީހަކު އޭނާގެ ޝަހުވަތް ފުއްދައިފި ނަމަ، އެ ކަމުގައި ވެސް އޭނާއަށް އަޖުރު ވޭތޯއެވެ؟ ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެ މީހެއްގެ ޝަހުވަތް ޙަރާމް ގޮތުގައި ފުއްދައިފި ނަމަ، އޭނާ ފާފަވެރިވާނެ ކަމަށް ތިޔަބައިމީހުން ނުދެކެމުހެއްޔެވެ؟ (އެބަހީ: ޝައްކެއް ވެސް ނެތެވެ.) ހަމަ އެފަދައިން، އެ މީހަކު ޙަލާލު ގޮތަކަށް ޝަހުވަތް ފުއްދައިފި ނަމަ، އޭނާއަށް އަޖުރު ވެސް ވެއެވެ.“

#26

[كُلُّ سُلَامَى مِنَ النَّاسِ عَلَيهِ صَدَقَةٌ]

[ކޮންމެ ހުޅަކަށް ޞަދަޤާތެއް ވެއެވެ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «كُلُّ سُلَامَى مِنَ النَّاسِ عَلَيهِ صَدَقَةٌ. كُلَّ يَومٍ تَطْلُعُ فِيهِ الشَّمْسُ: تَعْدِلُ بَينَ الْإِثْنَينِ صَدَقَةٌ. وَتُعِينُ الرَّجُلَ فِي دَابَّتِهِ فَتَحْمِلُهُ عَلَيهَا، أَو تَرْفَعُ لَهُ عَلَيهَا مَتَاعَهُ صَدَقَةٌ. وَالْكَلِمَةُ الطَّيِّبَةُ صَدَقَةٌ. وَكُلُّ خُطْوَةٍ تَمْشِيهَا إِلَى الصَّلَاةِ صَدَقَةٌ. وَتُمِيطُ الْأَذَى عَنِ الطَّرِيقِ صَدَقَةٌ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”އިރު އަރާ ކޮންމެ ދުވަހަކު، މީސްތަކުންގެ ކޮންމެ ހުޅަކަށް (ނުވަތަ ކަށިގަނޑަކަށް) ޞަދަޤާތެއް ވެއެވެ. (އެ ގޮތުން) ދެ މީހެއްގެ ދެމެދުގައި ޢަދުލުވެރިވުމަކީ ޞަދަޤާތެކެވެ. (އެބަހީ: ސުލްހަކުރުން، ނުވަތަ އިންސާފުން ޙުކުމް ކުރުމުން ސަވާބު ލިބޭނެއެވެ.) އަދި މީހަކަށް އެހީތެރިވެ ދިނުމުގެ ގޮތުން، އެ މީހާގެ ސަވާރީގެ މައްޗަށް އޭނާ އަރުވައި ދިނުން، ނުވަތަ އޭނާގެ ސާމާނު އެ ސަވާރީގެ މައްޗަށް އަރުވައި ދިނުމަކީ ޞަދަޤާތެކެވެ. އަދި ހެޔޮ ބަހަކީ ޞަދަޤާތެކެވެ. އަދި ނަމާދަށް ދާން އަޅާ ކޮންމެ ފިޔަވަޅަކީ ޞަދަޤާތެކެވެ. އަދި (މީސްތަކުންނަށް) އުނދަގޫވާ ފަދަ އެއްޗެއް މަގުމަތިން ނަގައި ދުރު ކުރުމަކީ ޞަދަޤާތެކެވެ.“

#27

[الْبِرُّ حُسْنُ الْخُلُقِ]

[ހެޔޮކަމަކީ އަޚްލާޤު ރަނގަޅުވުމެވެ]

عَنِ النَّوَّاسِ بْنِ سَمْعَانَ رَضِيَ اللَّهُ عَنْهُ عَنِ النَّبِيِّ ﷺ قَالَ: «الْبِرُّ: حُسْنُ الْخُلُقِ. وَالْإِثْمُ: مَا حَاكَ فِي نَفْسِكَ، وَكَرِهْتَ أنْ يَطَّلِعَ عليهِ النَّاسُ» رَوَاهُ مُسْلِمٌ.
وَعَنْ وَابِصَةَ بْنِ مَعْبَدٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: أَتَيتُ رَسُولَ اللَّهِ ﷺ فَقَالَ: «جِئْتَ تَسْأَلُ عَنِ الْبِرِّ وَالْإِثْمِ؟» قُلْتُ: نَعَمْ. قَالَ: «اسْتَفْتِ قَلْبَكَ، الْبِرُّ: مَا اطْمَأَنَّتْ إِلَيهِ النَّفْسُ، وَاطْمَأَنَّ إِلَيهِ الْقَلْبُ. وَالْإِثْمُ: مَا حَاكَ فِي النَّفْسِ، وَتَرَدَّدَ فِي الصَّدْرِ -وَإِنْ أَفْتَاكَ النَّاسُ، وَأَفْتَوكَ-» حَدِيثٌ حَسَنٌ، رُوِّينَاهُ فِي ’مُسْنَدَيِ الْإِمَامَينِ أَحْمَدَ بْنِ حَنْبَلٍ، والدَّارِمِيِّ‘ بِإِسْنَادٍ حَسَنٍ.

އައްނައްވާސް ބިން ސަމްޢާނު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހެޔޮކަމަކީ، އަޚްލާޤު ރަނގަޅުވުމެވެ. އަދި ފާފައަކީ، ހިތުގައި ނުތަނަވަސްކަން އުފައްދާ، އަދި (އެ ފާފައިގައި ތިބާ ވަނިކޮށް) އެހެން މީހުންނަށް ފެންނަން ތިބާ ނޭދޭ ކަންކަމެވެ.“
އަދި ވާބިޞާ ބިން މަޢުބަދު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ އަރިހަށް ތިމަން ދިޔުމުން ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިޔަ ދުރުވީ ހެޔޮ ކަންކަމާއި އަދި ފާފައާއި ބެހޭ ގޮތުން ސުވާލު ކުރައްވަންތޯއެވެ؟“ ތިމަން ދެންނެވީމެވެ: އާދޭހެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާގެ ހިތާ (ސުވާލުކޮށް) ފަތުވާ އަހާށެވެ. ހެޔޮކަމަކީ، ނަފްސު އެ ކަމަކާ މެދު ހަމަޖެހި، އަދި ހިތްވެސް އެ ކަމަކާ މެދު ހަމަޖެހިގެންވާ ކަންކަމެވެ. އަދި ފާފައަކީ، ނަފްސު އެ ކަމަކާ މެދު ނުތަނަވަސްވެ، އަދި ހިތް އެ ކަމަކާ މެދު ޖެހިލުންވާ ކަންކަމެވެ. އެ ނޫން ގޮތަކަށް ކިތަންމެ ވަރަކަށް މީހުން ފަތުވާދިން ކަމުގައި ވިޔަސްމެއެވެ.“

حَدِيثٌ حَسَنٌ، رُوِّينَاهُ فِي مُسْنَدَيِ الْإِمَامَينِ أَحْمَدَ بْنِ حَنْبَلٍ [18001، وَقَالَ الأَلبَانِي حَسَنٌ لِغَيرِهِ فِي صَحِيح التَّرغِيب وَالتَّرهِيب 1734]، والدَّارِمِيِّ [2575] بِإِسْنَادٍ حَسَنٍ.

[أُوصِيكُمْ بِتَقْوَى اللَّهِ وَالسَّمْعِ وَالطَّاعَةِ]

[ﷲ އަށް ތަޤުވާވެރިވެ، ވެރިމީހާގެ އަޑު އަހައި ކިޔަމަންވާށެވެ. އަދި ސުންނަތުގައި ވަރުގަދައަށް ހިފަހައްޓާށެވެ.]

عَنْ أَبِي نَجِيحٍ، الْعِرْبَاضِ بْنِ سَارِيَةَ رَضِيَ اللَّهُ عَنْهُ قَالَ: وَعَظَنَا رَسُولُ اللَّهِ ﷺ مَوعِظَةً وَجِلَتْ مِنْهَا الْقُلوبُ، وَذَرَفَتْ مِنْهَا الْعُيُونُ، فَقُلْنَا: يَا رَسُولَ اللَّهِ! كَأَنَّهَا مَوعِظَةُ مُوَدِّعٍ؛ فَأَوصِنَا. قَالَ: «أُوصِيكُمْ بِتَقْوَى اللَّهِ، وَالسَّمْعِ وَالطَّاعَةِ، وَإِنْ تَأَمَّرَ عَلَيكُمْ عَبْدٌ، فَإِنَّهُ مَنْ يَعِشْ مِنْكُمْ بَعْدِي فَسَيَرَى اخْتِلَافًا كَثِيرًا؛ فَعَلَيكُمْ بِسُنَّتِي، وَسُنَّةِ الْخُلَفَاءِ الرَّاشِدِينَ الْمَهْدِيِّينَ، عَضُّوا عَلَيهَا بِالنَّوَاجِذِ، وَإِيَّاكُمْ وَمُحْدَثَاتِ الْأُمُورِ، فَإِنَّ كُلَّ بِدْعَةٍ ضَلَالَةٌ»

އަބޫ ނަޖީޙު، އަލްޢިރްބާޟު ބިން ސާރިޔާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ތިމަންމެންނަށް (ވަރަށް އަސަރު ގަދަ) ވަޢުޡެއް ދެއްވިއެވެ. އެ ވަޢުޡުގެ ސަބަބުން ހިތްތައް ތެޅިގަނެ، ލޯތަކުން ކަރުނަ އޮހޮރިގެން ދިޔައެވެ. ދެން ތިމަންމެން ދެންނެވީމެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. ތިޔައީ ވަދާޢީ ވަޢުޡެއް ފަދައެވެ. ވީމާ ތިމަންމެންނަށް ވަޞިއްޔަތެއް ކޮށްދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިމަން ނަބިއްޔާ ތިޔަބައިމީހުންނަށް ވަޞިއްޔަތް ކުރައްވަނީ، ﷲ އަށް ތަޤުވާވެރިވުމަށާއި، އަދި ތިޔަބައިމީހުންގެ މައްޗަށް އަޅަކު ވެރިކަމަށް އައިކަމުގައި ވިޔަސް (އެ ވެރިމީހާގެ) އަޑުއަހައި ކިޔަމަންތެރިވުމަށެވެ. ފަހެ ހަމަކަށަވަރުން ތިމަން ނަބިއްޔާއަށް ފަހު، ތިޔަބައިމީހުންގެ ތެރެއިން ދިރިހުރި މީހަކަށް ގިނަގުނަ އިޚްތިލާފު ފެންނާނެއެވެ. ވީމާ، ތިޔަބައިމީހުންގެ މައްޗަށް އޮތީ ތިމަން ނަބިއްޔާގެ ސުންނަތުގައި ހިފެހެއްޓުމެވެ. އަދި ތެދުމަގު ލިބިގެންވާ ޚަލީފާއިންގެ ސުންނަތުގައި ހިފެހެއްޓުމެވެ. ތިޔަބައިމީހުން އެ ސުންނަތުގައި ކޮލުދަތްތަކުން ވަރުގަދައަށް ދަތް އަޅާ ހިފަހައްޓާށެވެ. އަދި (ދީނުގައި) އަލަށް އުފެދިގެން އަންނަ ކަންކަމާ (ބިދުޢަތަކާ) މެދު ސަމާލުވެ ރައްކާތެރިވާށެވެ. އެހެނީ ހަމަކަށަވަރުން، ކޮންމެ ބިދުޢައަކީ މަގުފުރެދުމެކެވެ.“

رَوَاهُ أَبُو دَاوُدَ [4607]، وَالتِّرْمِذِيُّ [2676]، وَقَالَ: ”حَدِيثٌ حَسَنٌ صَحِيحٌ“ [وَصَحَّحَهُ الأَلبَانِي فِي صَحِيح سُنَن التِّرمِذِي] [وَاللَفظُ لِلبَيهَقِي فِي سُنَنِهِ 20397]

[أَخْبِرْنِي بِعَمَلٍ يُدْخِلُني الْجَنَّةَ]

[ތިމަން ސުވަރުގެއަށް ވައްދައިދޭނެ ޢަމަލެއްގެ ޚަބަރު ދެއްވާށެވެ]

عَنْ مُعَاذِ بْنِ جَبَلٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: قُلْتُ: يَا رَسُولَ اللَّهِ! أَخْبِرْنِي بِعَمَلٍ يُدْخِلُني الْجَنَّةَ، وَيُبَاعِدُنِي مِنَ النَّارِ. قَالَ: «لَقَدْ سَأَلْتَ عَنْ عَظِيمٍ -وَإِنَّهُ لَيَسِيرٌ عَلَى مَنْ يَسَّرَهُ اللَّهُ عَلَيهِ-: تَعْبُدُ اللَّهَ لَا تُشْرِكُ بِهِ شَيئًا، وَتُقِيمُ الصَّلَاةَ، وَتُؤْتِي الزَّكَاةَ، وَتَصُومُ رَمَضَانَ، وَتَحُجُّ الْبَيتَ» ثُمَّ قَالَ: «أَلَا أَدُلُّكَ عَلَى أَبْوَابِ الْخَيرِ؟ الصَّومُ جُنَّةٌ، وَالصَّدَقَةُ تُطْفِئُ الْخَطِيئَةَ كَمَا يُطْفِئُ الْمَاءُ النَّارَ، وَصَلاَةُ الرَّجُلِ مِنْ جَوفِ اللَّيلِ» ثُمَّ تَلَا: ﴿تَتَجَافَى جُنُوبُهُمْ عَنِ الْمَضَاجِعِ﴾ حَتَّى بَلَغَ: ﴿يَعْمَلُونَ﴾ ثُمَّ قَالَ: «أَلَا أُخْبِرُكَ بِرَأْسِ الْأَمْرِ، وَعَمُودِهِ، وَذِرْوَةِ سَنَامِهِ؟» قُلْتُ: بَلَى يَا رَسُولَ اللَّهِ! قَالَ: «رَأْسُ الْأَمْرِ: الْإِسْلاَمُ، وَعَمُودُهُ: الصَّلَاةُ، وَذِرْوَةُ سَنَامِهِ: الْجِهَادُ» ثُمَّ قَالَ: «أَلَا أُخْبِرُكَ بِمِلَاكِ ذَلِكَ كُلِّهِ؟» قُلْتُ: بَلَى يَا رَسُولَ اللَّهِ! فَأَخَذَ بِلِسَانِهِ، وَقَالَ: «كُفَّ عَلَيكَ هَذَا» قُلْتُ: يَا نَبِيَّ اللَّهِ! وَإِنَّا لَمُؤَاخَذُونَ بِمَا نَتَكَلَّمُ بِهِ؟! فَقَالَ: «ثَكِلَتْكَ أُمُّكَ، وَهَلْ يَكُبُّ النَّاسَ فِي النَّارِ عَلَى وُجُوهِهِمْ -أَو عَلَى مَنَاخِرِهِمْ- إِلَّا حَصَائِدُ أَلْسِنَتِهِمْ؟!»

މުޢާޛު ބިން ޖަބަލު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ތިމަން ދެންނެވީމެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. ތިމަން ސުވަރުގެއަށް ވައްދައިދޭނެ، އަދި ނަރަކައިން ދުރުކުރެވޭނެ ޢަމަލެއްގެ ޚަބަރު ތިމަންނާއަށް ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ތިޔަ ސުވާލު ކުރެއްވީ ބޮޑުވެގެންވާ ކަމަކާ ބެހޭ ގޮތުންނެވެ. އަދި ހަމަކަށަވަރުން ﷲ ފަސޭހަ ކޮށްދެއްވައިފި މީހަކަށް، އެ ކަމަކީ ނުހަނު ފަސޭހަ ކަމަކަށް ވެގެންދެއެވެ. ﷲ އާ އެއްވެސް ފަރާތެއް ޝަރީކު ކުރުމަކާ ނުލައި، ހަމައެކަނި އެ އިލާހަށް އަޅުކަން ކުރާށެވެ. އަދި ނަމާދު ގާއިމުކޮށް، (މުދަލު) ޒަކާތް ދީ، ރަމަޟާން މަހު ރޯދަ ހިފައި، ގެފުޅަށް ގޮސް ޙައްޖު ވާށެވެ.“ ދެން ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިމަން ނަބިއްޔާ، ތިބާއަށް ހެޔޮކަމުގެ ދޮރުތައް ދައްކަވައި ދެއްވަންތޯއެވެ؟ ރޯދައަކީ އައްޑަނައެކެވެ. (އެބަހީ: ރައްކާތެރި ކަމެކެވެ.) އަދި ފެނުން އަލިފާން ނިވޭ ފަދައިން، ޞަދަޤާތުން ފާފަ ފޮހެވިގެންދެއެވެ. އަދި (ހަމަ އެފަދައިން) މީހަކު ރޭގަނޑުގެ މެންދަމު ކުރާ ނަމާދު ވެސްމެއެވެ.“ ދެން ނަބިއްޔާ (މި އާޔަތް) ކިޔަވައި ވިދާޅުވިއެވެ: ❞އެބައިމީހުންގެ އަރިކަށިތައް ތަންމަތިތަކާ ފޭވެ (ރޭއަޅުކަން ކުރުމަށް ޓަކައި ނިދި ދޫކޮށްލާފައި ތެދުވެ) އެބައިމީހުންގެ ރައްބަށް ބިރުވެތިކަމާއި އުންމީދާއެކު އެބައިމީހުން ދުޢާ ކުރެއެވެ. އަދި ތިމަން އިލާހު އެބައިމީހުންނަށް ދެއްވި ތަކެތިން، އެބައިމީހުން ހޭދަ ކުރެއެވެ. ފަހެ އެބައިމީހުންނަށް ޓަކައި ފޮރުއްވާފައިވާ ލޮލުގެ ފިނިކަން (އެބަހީ: ސުވަރުގޭގެ ނިޢުމަތްތައް) އެއްވެސް ނަފްސަކަށް ނޭނގޭނެއެވެ. އެއީ އެބައިމީހުން ކޮށްއުޅުނު ޢަމަލުތަކުގެ ޖަޒާއެއްގެ ގޮތުންނެވެ.❝ ދެން ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މި ދީނުގެ އެންމެ މުހިންމު ކަމަކީ ކޮބައިކަމާއި، މި ދީނުގެ ތަނބާއި ކުރިބޯއްޓާ ބެހޭ ގޮތުން ތިބާއަށް ކިޔައި ދެއްވަންހެއްޔެވެ؟“ ތިމަން ދެންނެވީމެވެ: އާދޭހެވެ. އޭ ﷲގެ ރަސޫލާއެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މި ދީނުގެ އެންމެ މުހިންމު ކަމަކީ އިސްލާމްކަމެވެ. އަދި މި ދީނުގެ ތަނބަކީ ނަމާދެވެ. އަދި މި ދީނުގެ ކުރިބޯއްޓަކީ ޖިހާދެވެ.“ އެއަށްފަހު ޙަދީޘް ކުރެއްވިއެވެ: ”އެ ހުރިހާ ކަމަކަށް ބާރުވެރިކޮށްދޭނެ ކަމަކާ ބެހޭ ގޮތުން ތިބާއަށް ކިޔައި ދެއްވަންހެއްޔެވެ؟“ ތިމަން ދެންނެވީމެވެ: އާދޭހެވެ. އޭ ﷲގެ ރަސޫލާއެވެ. ދެން ނަބިއްޔާ ދޫފުޅުގައި ހިފަހައްޓަވާފައި ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާ މީތި ހިފަހައްޓާށެވެ.“ ތިމަން ދެންނެވީމެވެ: އޭ ﷲގެ ނަބިއްޔާއެވެ. ދުލުން ދައްކާ ވާހަކަތަކުގެ ސަބަބުން ތިމަންމެން ހިއްޕެވޭނެތޯއެވެ؟ (އެބަހީ: ޢުޤޫބާތް ދެވޭނެތޯއެވެ؟) ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާގެ މަންމައަށް ތިބާ ގެއްލިދާށެވެ. (ނޯޓު: މިއީ ކަމެއްގެ ބޮޑުކަން ހާމަކުރަން ޢަރަބީންގެ ދޫމަތިން ބުނެ އުޅޭ އިބާރާތެކެވެ. ބަދު ދުޢާއެއް ނޫނެވެ.) ފަހެ މީހުންގެ ދޫތަކުން ހޯދައިގަތް ކަންތަކުން މެނުވީ، އެބައިމީހުން ނަރަކައިގެ ތެރެއަށް މޫނުވަތަށް -ނުވަތަ ނޭފަތް ވަތަށް- އެއްލާލެވޭނެހެއްޔެވެ؟“

رَوَاهُ التِّرْمِذِيُّ [2616]، وَقَالَ: ”حَدِيثٌ حَسَنٌ صَحِيحٌ“ [وَصَحَّحَهُ الأَلبَانِي فِي صَحِيح سُنَن التِّرمِذِي]

[إِنَّ اللَّهَ فَرَضَ فَرَائِضَ فَلَا تُضَيِّعُوهَا]

[ﷲ ފަރުޟު ކުރެއްވި ކަންތައްތައް ދޫކޮށް ނުލާށެވެ]

عَنْ أَبِي ثَعْلَبَةَ الْخُشَنِيِّ- جُرْثُومِ بْنِ نَاشِرٍ- رَضِيَ اللَّهُ عَنْهُ عَنْ رَسُولِ اللَّهِ ﷺ قَالَ: «إِنَّ اللَّهَ فَرَضَ فَرَائِضَ فَلَا تُضَيِّعُوهَا، وَحَدَّ حُدُودًا فَلَا تَعْتَدُوهَا، وَحَرَّمَ أَشْيَاءَ فَلَا تَنْتَهِكُوهَا، وَسَكَتَ عَنْ أَشْيَاءَ -رَحْمَةً لَكُمْ غَيرَ نِسْيَانٍ- فَلَا تَبْحَثُوا عَنْهَا»

އަބޫ ޘަޢުލަބާ އަލްޚުޝަނީ، ޖުރްޘޫމް ބިން ނާޝިރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ﷲ ވަނީ ފަރުޟު ކަންތައްތަކެއް ކަނޑައަޅުއްވާފައެވެ. ވީމާ ތިޔަބައިމީހުން އެ ކަންތައްތަކަށް (އިހުމާލުވެ) ދޫކޮށް ނުލާށެވެ. އަދި އެ އިލާހު (އިންތަކަކާއި) ހައްދުތަކެއް ކަނޑައެޅުއްވިއެވެ. ވީމާ ތިޔަބައިމީހުން އެ ހައްދުތައް ފަހަނަޅައި ނުދާށެވެ. އަދި އެ އިލާހު ކަންތައްތަކެއް ޙަރާމް ކުރެއްވިއެވެ. ވީމާ ތިޔަބައިމީހުން އެ ކަންތައްތަކަށް އަރައި ނުގަންނާށެވެ. އަދި އެ އިލާހު (ބައެއް) ކަންތައްތަކާ މެދު ސުކޫތުވެވޮޑިގަތެވެ. (އެބަހީ: ވަކި ގޮތެއް ކަނޑަނާޅުއްވައި ދޫކޮށްލެއްވިއެވެ.) އެއީ ތިޔަބައިމީހުންނަށް ރަޙުމަތެއްގެ ގޮތުންނެވެ. ހަނދުމަނެތިވެ ވޮޑިގެނެއް ނޫނެވެ. ވީމާ ތިޔަބައިމީހުން އެ ކަންތައްތައް ހޯދަން ނޫޅޭށެވެ.“

#31

[ازْهَدْ فِي الدُّنْيَا يُحِبَّكَ اللَّهُ]

[ދުނިޔެއާމެދު ޒުހުދުވެރިވޭ، އޭރުން ﷲ ތިބާ ދެކެ ލޯބިވެ ވޮޑިގަންނަވާނެއެވެ]

عَنْ أَبِي الْعَبَّاسِ، سَهْلِ بْنِ سَعْدٍ السَّاعِدِيِّ رَضِيَ اللَّهُ عَنْهُ قَالَ: جَاءَ رَجُلٌ إِلَى النَّبيِّ ﷺ فَقَالَ: يَا رَسُولَ اللَّهِ! دُلَّنِي عَلَى عَمَلٍ إِذَا عَمِلْتُهُ أَحَبَّنِيَ اللَّهُ، وَأَحَبَّنِيَ النَّاسُ. فَقَالَ: «ازْهَدْ فِي الدُّنْيَا يُحِبَّكَ اللَّهُ، وَازْهَدْ فِيمَا عِنْدَ النَّاسِ يُحِبَّكَ النَّاسُ»

އަބުލް ޢައްބާސް، ސަހްލު ބިން ޞަޢުދު އައްސާޢިދީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ގެ އަރިހަށް މީހަކު އައިސް ދެންނެވިއެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. އެ ކަމެއް ކުރުމުން، ތިމަން ދެކެ ﷲ ލޯބިވެ ވޮޑިގަންނަވާނެ، އަދި މީސްތަކުން ވެސް ތިމަން ދެކެ ލޯބިވާނެ ޢަމަލެއް ދައްކަވައި ދެއްވާށެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިބާ ދުނިޔޭގައި ޒުހުދުވެރިވާށެވެ. (އެބަހީ: ދުނިޔެއަށް އެދުންވެރި ނުވާށެވެ.) އޭރުން ﷲ ތިބާ ދެކެ ލޯބިވެ ވޮޑިގަންނަވާނެއެވެ. އަދި މީހުންގެ (އަތުގައިވާ) ތަކެއްޗާމެދު ޒުހުދުވެރިވާށެވެ. އޭރުން ތިބާ ދެކެ މީހުން ލޯބިވާނެއެވެ.“

حَدِيثٌ حَسَنٌ، رَوَاهُ ابْنُ مَاجَهْ [4102]، وَغَيرُهُ، بِأَسَانِيدَ حَسَنَةٍ [وَقَالَ الأَلبَانِي صَحِيحٌ بِشَوَاهِدِهِ فِي تَحقِيق رِيَاض الصَّالِحِين 476]

[لاَ ضَرَرَ وَلاَ ضِرَارَ]

[ގެއްލުން ދިނުމެއް ވެސް އަދި ގެއްލުން ލިބިގަތުމެއް ވެސް އޮވެގެން ނުވާނެއެވެ]

عَنْ أَبِي سَعِيدٍ، سَعْدِ بْنِ مَالِكِ بْنِ سِنَانٍ الْخُدْرِيِّ رَضِيَ اللَّهُ عَنْهُ أَنَّ رَسُولَ اللَّهِ ﷺ قَالَ: «لاَ ضَرَرَ وَلاَ ضِرَارَ»

އަބޫ ސަޢީދު، ސަޢުދު ބިން މާލިކު ބިން ސިނާނު އަލްޚުޟްރީ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ގެއްލުން ދިނުމެއް އޮވެގެން ނުވާނެއެވެ. އަދި (މީހަކު ގެއްލުމެއް ދިނުމުން، ނުޙައްޤުން) ބަދަލުގައި ގެއްލުން ދިނުމެއް ވެސް އޮވެގެން ނުވާނެއެވެ.“ (ނޯޓު: އަދި މީގެ ތެރޭގައި މި ކަންތައް ވެސް ހިމެނޭނެއެވެ: ގެއްލުން ދިނުމެއް ވެސް އަދި ގެއްލުން ލިބިގަތުމެއް ވެސް ނުވެއެވެ. އަދި ގަސްދެއް ނެތި އަދި ގަސްދަކާލައިގެން ވިޔަސް ގެއްލުން ދިނުމެއް ނުވެއެވެ. އަދި އަމިއްލަ ނަފްސަށް ވިޔަސް އަދި އެހެން މީހަކަށް ވިޔަސް ގެއްލުން ދިނުމެއް ނުވެއެވެ.)

حَدِيثٌ حَسَنٌ، رَوَاهُ ابْنُ مَاجَهْ [2341]، وَالدَّارَقُطْنِيُّ [3079]، وَغَيرُهُمَا، مُسْنَدًا. وَرَوَاهُ مَالِكٌ فِي الْمُوَطَّأِ [2171] -عَنْ عَمْرِو بْنِ يَحْيَى، عَنْ أَبِيهِ، عَنِ النَّبِيِّ ﷺ- مُرْسَلًا، فَأَسْقَطَ أَبَا سَعِيدٍ. وَلَهُ طُرُقٌ يُقَوِّي بَعْضُهَا بَعْضًا [وَصَحَّحَهُ الأَلبَانِي فِي إِروَاء الغَلِيل فِي تَخرِيج أَحَادِيث مَنَار السَّبِيل 896]

[الْبَيِّنَةُ عَلَى الْمُدَّعِي]

[ހެކި ހުށަހަޅައި ސާބިތުކުރުން ވަނީ ދަޢުވާ ކުރާ މީހާގެ މައްޗަށެވެ]

عَنِ ابْنِ عَبَّاسٍ رَضِيَ اللَّهُ عَنْهُمَا أَنَّ رَسُولَ اللَّهِ ﷺ قَالَ: «لَو يُعْطَى النَّاسُ بِدَعْوَاهُمْ، لَادَّعَى رِجَالٌ أَمْوَالَ قَومٍ وَدِمَاءَهُمْ، لَكِنِ الْبَيِّنَةُ عَلَى الْمُدَّعِي، وَالْيَمِينُ عَلَى مَنْ أَنْكَرَ»

އިބްނު ޢައްބާސް رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމައެކަނި މީސްތަކުން ކޮށްލާ ދަޢުވާއަކުން، (ޝަރީޢަތުން) އެބައިމީހުންނަށް ކަންކަން ނިންމައިދެވޭ ނަމަ، (ﷲ އަށް ބިރުވެތި ނުވާ) މީހުން، އެހެން މީހުންގެ މުދަލާއި (މަރުގެ) ދިޔައަށް ދަޢުވާ ކުރީސްކަން ކަށަވަރެވެ. އެހެނެއްކަމަކު، ދަޢުވާ ކުރާ މީހާގެ މައްޗަށް ވަނީ (ދަޢުވާގެ ތެދުކަން ސާބިތު ކުރުމަށް) ހެކި ހުށަހެޅުމެވެ. އަދި (ދަޢުވާ ލިބި، އެއަށް) އިންކާރު ކުރާ މީހާގެ މައްޗަށް ވަނީ (އެ ދަޢުވާގެ ދޮގުކަން ބަޔާންކޮށް، ﷲ ގަންދީ) ހުވާ ކުރުމެވެ.“ (އެބަހީ: މީހެއްގެ މައްޗަށް ކުރެވޭ ދަޢުވާ ސާބިތު ކުރެވިއްޖައުމަށް ދާންދެން، އޭނާ ވާނީ އެ ކަމަކުން ބަރީއަ ވެގެންނެވެ.)

#34

[مَنْ رَأَى مِنْكُمْ مُنْكَرًا فَلْيُغَيِّرْهُ بِيَدِهِ]

[ނުބައި ކަމެއް ފެނިއްޖެ ނަމަ އެ ކަން އަތުން ބަދަލު ކުރާށެވެ]

عَنْ أَبِي سَعِيدٍ الْخُدْرِيِّ رَضِيَ اللَّهُ عَنْهُ قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «مَنْ رَأَى مِنْكُمْ مُنْكَرًا فَلْيُغَيِّرْهُ بِيَدِهِ، فَإِنْ لَمْ يَسْتَطِعْ فَبِلِسَانِهِ، فَإِنْ لَمْ يَسْتَطِعْ فَبِقَلْبِهِ، وَذَلِكَ أَضْعَفُ الْإِيمَانِ»

އަބޫ ސަޢީދު އަލްޚުދްރީ رَضِي اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”ތިޔަބައިމީހުންގެ ތެރެއިން މީހަކަށް ނުބައިކަމެއް ފެނިއްޖެ ނަމަ، ފަހެ އޭނާގެ އަތުން އެ ކަމެއް ބަދަލު ކުރާށެވެ. ދެން (އެ ކަމަށް) ކުޅަދާނަވެގެން ނުވާ ނަމަ، ފަހެ އޭނާގެ ދުލުންނެވެ. ދެން (އެ ކަމަށް) ވެސް ކުޅަދާނަވެގެން ނުވާ ނަމަ، ފަހެ އޭނާގެ ހިތުންނެވެ. އަދި އެއީ އީމާންކަމުގެ އެންމެ ބަލިކަށި ދަރަޖައެވެ.“ (އެބަހީ: ފާފައެއް ކުރާތަން ފެނިއްޖެ ނަމަ، ޢަމަލީ ގޮތުން އެ ކަން ހުއްޓުވާށެވެ. ނުކުރެވޭ ނަމަ، ދެން އަނގަބަހުން ނުވަތަ ލިޔުމުން އެ ކަން ކުށްވެރިކޮށް، އޭގެ ނުބައިކަން ބަޔާން ކުރާށެވެ. އެވަރު ވެސް ނުކުރެވޭ ނަމަ، ދެން އެ ކަމަށް ނަފުރަތުކޮށް، އެ ކަން ހުއްޓުމަށް ހިތުން އެދޭށެވެ.)

#35

[الْمُسْلِمُ أَخُو الْمُسْلِمِ]

[އެއް މުސްލިމަކީ އަނެއް މުސްލިމެއްގެ އަޚެކެވެ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «لَا تَحَاسَدُوا، وَلَا تَنَاجَشُوا، وَلَا تَبَاغَضُوا، وَلَا تَدَابَرُوا، وَلَا يَبِعْ بَعْضُكُمْ عَلَى بَيعِ بَعْضٍ، وَكُونُوا عِبَادَ اللَّهِ إِخْوَانًا. الْمُسْلِمُ أَخُو الْمُسْلِمِ، لَا يَظْلِمُهُ، وَلَا يَخْذُلُهُ، وَلَا يَحْقِرُهُ، التَّقْوَى هَهُنَا -وَيُشِيرُ إِلَى صَدْرِهِ ثَلاَثَ مَرَّاتٍ-. بِحَسْبِ امْرِئٍ مِنَ الشَّرِّ أَنْ يَحْقِرَ أَخَاهُ الْمُسْلِمَ. كُلُّ الْمُسْلِمِ عَلَى الْمُسْلِمِ حَرَامٌ: دَمُهُ، وَمَالُهُ، وَعِرْضُهُ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ތިޔަބައިމީހުން އެއްބަޔަކު އަނެއްބަޔަކާ ހަސަދަވެރި ނުވާށެވެ. އަދި އެއްބަޔަކު އަނެއްބަޔަކަށް (ލަނޑުދިނުމުގެ ގޮތުން ރާވައިގެން، އެއްޗެއް ގަތުމުގެ ގަސްދަކާ ނުލައި އޭގެ) އަގު ބޮޑުކޮށް ނުހަދާށެވެ. އަދި އެއްބަޔަކު އަނެއްބަޔަކަށް ނަފުރަތު ނުކުރާށެވެ. އަދި އެއްބަޔަކު އަނެއްބަޔަކަށް ފުރަގަސް ނުދޭށެވެ. އަދި އެއްބަޔަކު އަނެއްބަޔެއްގެ ވިޔަފާރީގެ މައްޗަށް ވިޔަފާރި ނުކުރާށެވެ. (އެބަހީ: ގަނެވިއްކުމުގައި، އެކަކު އަގު ކިޔަމުން ދަނިކޮށް، އަނެކަކު އެ ދޭތެރެއަށް ވަދެ އަގު ކިޔާ ނުހަދާށެވެ.) އަދި ތިޔަބައިމީހުން ﷲގެ އަޅުންގެ ގޮތުގައި، އަޚުންތަކެއް ކަމުގައި ވާށެވެ. އެއް މުސްލިމަކީ އަނެއް މުސްލިމެއްގެ އަޚެކެވެ. އޭނާގެ އަޚާއަށް އަނިޔާވެރިއެއް ނުވާނެއެވެ. އަދި (އޭނާއަށް އެހީނުވެ، ނިކަމެތިވާން) އޭނާ ދޫކޮށެއް ނުލާނެއެވެ. އަދި އޭނާއަށް އިހާނާތްތެރިވެ ނިކަމެތިކޮށެއް ނުހިތާނެއެވެ. ތަޤުވާވެރިކަން ވަނީ މި ތަނުގައެވެ. -ދެން (މި ފަދައިން ޙަދީޘް ކުރައްވާފައި) ނަބިއްޔާގެ މޭފުޅާ ދިމާޔަށް ތިން ފަހަރު އިޝާރާތް ކުރެއްވިއެވެ- (އަދި ޙަދީޘް ކުރެއްވިއެވެ.) މީހަކަށް ކުރެވޭ ނުބައި ކަމެއްގެ ގޮތުގައި، އޭނާގެ މުސްލިމު އަޚާއަށް އިހާނާތްތެރިވެ ނިކަމެތިކޮށްލުން ފުދެއެވެ. މުސްލިމެއްގެ ހުރިހާ އެއްޗެއް (ޙައްޤަކާ ނުލައި)، އަނެއް މުސްލިމަކަށް ޙަރާމްވެގެން ވެއެވެ. (އެ ގޮތުން) އޭނާގެ ލެޔާއި، އޭނާގެ މުދަލާއި، އަދި އޭނާގެ އަބުރު ވެސްމެއެވެ.“

#36

[مَنْ نَفَّسَ عَنْ مُؤْمِنٍ كُرْبَةً]

[މުއުމިނެއްގެ ކިބައިން ދުނިޔޭގެ ދަތިކަމެއް ފިލުވައިދީފި މީހާ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ عَنِ النَّبِيِّ ﷺ قَالَ: «مَنْ نَفَّسَ عَنْ مُؤْمِنٍ كُرْبَةً مِنْ كُرَبِ الدُّنْيَا؛ نَفَّسَ اللَّهُ عَنْهُ كُرْبَةً مِنْ كُرَبِ يَومِ الْقِيَامَةِ. وَمَنْ يَسَّرَ عَلَى مُعْسِرٍ؛ يَسَّرَ اللَّهُ عَلَيهِ فِي الدُّنْيَا وَالْآخِرَةِ. وَمَنْ سَتَرَ مُسْلِمًا؛ سَتَرَهُ اللَّهُ فِي الدُّنْيَا وَالْآخِرَةِ. وَاللَّهُ فِي عَونِ الْعَبْدِ مَا كَانَ الْعَبْدُ فِي عَونِ أَخِيهِ. وَمَنْ سَلَكَ طَرِيقًا يَلْتَمِسُ فِيهِ عِلْمًا؛ سَهَّلَ اللَّهُ لَهُ بِهِ طَرِيقًا إِلَى الْجَنَّةِ. وَمَا اجْتَمَعَ قَومٌ فِي بَيتٍ مِنْ بُيُوتِ اللَّهِ يَتْلُونَ كِتَابَ اللَّهِ، وَيَتَدَارَسُونَهُ بَينَهُمْ؛ إِلاَّ نَزَلَتْ عَلَيهِمُ السَّكِينَةُ، وَغَشِيَتْهُمُ الرَّحْمَةُ، وَحَفَّتْهُمُ الْمَلَائِكَةُ، وَذَكَرَهُمُ اللَّهُ فِيمَنْ عِنْدَهُ. وَمَنْ بَطَّأَ بِهِ عَمَلُهُ؛ لَمْ يُسْرِعْ بِهِ نَسَبُهُ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”މުއުމިނެއްގެ ކިބައިން ދުނިޔޭގެ ހިތާމަތަކުން ހިތާމައެއް ފިލުވައިދީފި މީހަކަށް، ޤިޔާމަތް ދުވަހުގެ ހިތާމަތަކުން ހިތާމައެއް ﷲ އޭނާގެ ކިބައިން ފިއްލަވައިދެއްވާނެއެވެ. އަދި ދަތި ހާލުގައި ޖެހިފައިވާ ފަރާތަކަށް ފަސޭހަ ކޮށްދީފި މީހަކަށް، ދުނިޔެއާއި އާޚިރަތުގައި ވެސް ﷲ އޭނާއަށް ފަސޭހަ ކޮށްދެއްވާނެއެވެ. އަދި މުސްލިމަކު ފަރުދާކޮށް (އޭނާގެ ކުށްތައް ފޮރުވައިދީފި) މީހަކަށް، ދުނިޔެއާއި އާޚިރަތުގައި ﷲ އޭނާ ފަރުދާކޮށްދެއްވައި (އޭނާގެ ކުށްތައް ފޮރުއްވައި ދެއްވާނެއެވެ.) އަދި (ﷲގެ) އަޅަކު، އޭނާގެ އަޚާއަށް އެހީތެރިވުމުގައި ވާހައި ހިނދަކު، ﷲ އޭނާއަށް އެހީތެރިވުމުގައި ވޮޑިގެންވާނެއެވެ. އަދި (ޝަރުޢީ) ޢިލްމު ހޯދުމަށް ޓަކައި މަގަކުން ހިނގައިފި މީހަކަށް، އެ ކަމުގެ ސަބަބުން ސުވަރުގެއަށް ދާނެ މަގެއް، ﷲ އޭނާއަށް ފަސޭހަ ކޮށްދެއްވާނެއެވެ. އަދި ﷲގެ ގެކޮޅުތަކުން ކުރެ ގެކޮޅެއްގައި (އެބަހީ: މިސްކިތެއްގައި) ބަޔަކު އެއްވެ، ﷲގެ ފޮތް އެބައިމީހުންގެ މެދުގައި ކިޔަވައި އުނގެނި ދަސްކުރާ ނަމަ، އެބައިމީހުންގެ މައްޗަށް ސަކީނާ (އެބަހީ: މަޑުމޮޅިކަމާއި ހިތްހަމަޖެހުން) ބާވައިނުލައްވާ ނުދާނެއެވެ. އަދި ރަޙުމަތުން އެބައިމީހުން ވަށާލެވޭނެއެވެ. އަދި މަލާއިކަތުން ވެސް އެބައިމީހުން ވަށާލައްވާނެއެވެ. އަދި ﷲގެ ޙަޟްރަތުގައިވާ މަލާއިކަތުންގެ އަރިހުގައި، އެ އިލާހު އެބައިމީހުންގެ (ވާހަކަ) ޒިކުރު ކުރައްވާނެއެވެ. އަދި އެ މީހެއްގެ (ހެޔޮ) ޢަމަލު ލަސްކޮށް (ފަރުވާ ކުޑަކޮށްފި) މީހަކު، އެ މީހެއްގެ ނަސަބުގެ މޮޅުކަމުން އޭނާއަށް ލިބޭނެ އެއްވެސް ކުރިއެރުމެއް ނޯންނާނެއެވެ.“

#37

[إِنَّ اللَّهَ كَتَبَ الْحَسَنَاتِ وَالسَّيِّئَاتِ]

[ﷲ ވަނީ ހެޔޮ ކަންތަކާއި ނުބައި ކަންތައް ލިޔުއްވާފައެވެ]

عَنِ ابْنِ عَبَّاسٍ رَضِيَ اللَّهُ عَنْهُمَا عَنْ رَسُولِ اللَّهِ ﷺ -فِيمَا يَرْوِيهِ عَنْ رَبِّهِ تَبَارَكَ وَتَعالَى- قَالَ: «إِنَّ اللَّهَ كَتَبَ الْحَسَنَاتِ وَالسَّيِّئَاتِ، ثُمَّ بَيَّنَ ذَلِكَ: فَمَنْ هَمَّ بِحَسَنَةٍ فَلَمْ يَعْمَلْهَا؛ كَتَبَهَا اللَّهُ عِنْدَهُ حَسَنَةً كَامِلَةً. وَإِنْ هَمَّ بِهَا فَعَمِلَهَا؛ كَتَبَهَا اللَّهُ عِنْدَهُ عَشْرَ حَسَنَاتٍ إِلَى سَبْعِ مِائَةِ ضِعْفٍ إِلَى أَضْعَافٍ كَثِيرَةٍ. وَإِنْ هَمَّ بِسَيِّئَةٍ فَلَمْ يَعْمَلْهَا؛ كَتَبَهَا اللَّهُ عِنْدَهُ حَسَنَةً كَامِلَةً. وَإِنْ هَمَّ بِهَا فَعَمِلَهَا؛ كَتَبَهَا اللَّهُ سَيِّئَةً وَاحِدَةً»

ޢަބްދު ﷲ ބިން ޢައްބާސް رَضِيَ اللَّهُ عَنْهُمَا ރިވާ ކުރެއްވިއެވެ: ﷲ ތަބާރަކަ ވަތަޢާލާގެ ޙަޟްރަތުން، ރަސޫލު ﷲ ﷺ ރިވާ ކުރެއްވި (ޤުދުސީ ޙަދީޘެއްގައި ވެއެވެ.) ”ހަމަކަށަވަރުން ﷲ ވަނީ (އިންސާނާ ކުރާ) ހެޔޮ ކަންތަކާއި ނުބައި ކަންތައް ލިޔުއްވާފައެވެ. ދެން އެ ކަންތައް ބަޔާން ކޮށްދެއްވިއެވެ: ފަހެ މީހަކު ހެޔޮ ކަމެއް ކުރުމަށް ގަސްދު ކުރުމަށް ފަހު އެކަން ނުކޮށްފި ނަމަ، ﷲގެ ޙަޟްރަތުގައި އެކަން އެއީ ފުރިހަމަ ހެޔޮ ކަމެއްގެ ގޮތުގައި އެ އިލާހު ލިޔުއްވައެވެ. އަދި އެ މީހާ ހެޔޮ ކަމެއް ކުރުމަށް ގަސްދުކޮށް އެ ކަންކޮށްފި ނަމަ، ﷲގެ ޙަޟްރަތުގައި އެ ކަން އެއީ ދިހަ ހެޔޮ ކަމުން ފެށިގެން ހަތް ސަތޭކަ ގުނައަށް، ނުވަތަ އެއަށް ވުރެ ވެސް އިތުރު އެތައް ގުނައަކަށް އެ އިލާހު ލިޔުއްވައެވެ. އަދި އެ މީހާ ނުބައި ކަމެއް ކުރުމަށް ގަސްދުކުރުމަށް ފަހު، އެކަން ނުކޮށް ދޫކޮށްލައިފި ނަމަ، ﷲގެ ޙަޟްރަތުގައި އެ ކަން އެއީ ފުރިހަމަ ހެޔޮ ކަމެއްގެ ގޮތުގައި އެ އިލާހު ލިޔުއްވައެވެ. އަދި އެ މީހާ ނުބައި ކަމެއް ކުރުމަށް ގަސްދުކޮށް އެކަންކޮށްފި ނަމަ، ﷲގެ ޙަޟްރަތުގައި އެކަން އެއީ އެންމެ ނުބައި ކަމެއްގެ ގޮތުގައި އެ އިލާހު ލިޔުއްވައެވެ.“

#38

[مَنْ عَادَى لِي وَلِيًّا]

[ތިމަން އިލާހުގެ ވަލީވެރިއަކާ އަދާވާތްތެރިވެއްޖެ މީހާ]

عَنْ أَبِي هُرَيرَةَ رَضِيَ اللَّهُ عَنْهُ قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: «إِنَّ اللَّهَ تَعَالَى قَالَ: مَنْ عَادَى لِي وَلِيًّا فَقَدْ آذَنْتُهُ بِالْحَرْبِ، وَمَا تَقَرَّبَ إِلَيَّ عَبْدِي بِشَيءٍ أَحَبَّ إِلَيَّ مِمَّا افْتَرَضْتُ عَلَيهِ. وَمَا يَزَالُ عَبْدِي يَتَقَرَّبُ إِلَيَّ بِالنَّوَافِلِ حَتَّى أُحِبَّهُ، فَإِذَا أَحْبَبْتُهُ كُنْتُ سَمْعَهُ الَّذِي يَسْمَعُ بِهِ، وَبَصَرَهُ الَّذِي يُبْصِرُ بِهِ، وَيَدَهُ الَّتِي يَبْطِشُ بِهَا، وَرِجْلَهُ الَّتِي يَمْشِي بِهَا، وَلَئِنِ سَأَلَنِي لَأُعْطِيَنَّهُ، وَلَئِنِ اسْتَعَاذَنِي لَأُعِيذَنَّهُ»

އަބޫ ހުރައިރާ رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: (ޤުދުސީ ޙަދީޘެއްގައި) ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ﷲ ތަޢާލާ ވަޙީ ކުރެއްވިއެވެ: ތިމަން އިލާހުގެ ވަލީވެރިއަކާ (އެބަހީ: ތިމަން އިލާހުގެ ގާތްކަން ހާސިލުކޮށްފައިވާ ތަޤުވާވެރި އަޅަކާ) އަދާވާތްތެރި ވެއްޖެ މީހާ (ދަންނާށެވެ!) ފަހެ، ތިމަން އިލާހު އޭނާއާ ދެކޮޅަށް ހަނގުރާމަ އިޢުލާން ކުރައްވައިފީމުކަން ކަށަވަރެވެ. އަދި ތިމަން އިލާހުގެ އަޅާގެ މައްޗަށް ފަރުޟު ކުރެއްވި ކަންތައްތައް އަދާ ކުރުމަށް ވުރެ، ތިމަން އިލާހުގެ ޙަޟްރަތުގައި ލޮބުވެތިކަން ބޮޑު ކަމަކުން އެ އަޅާ ތިމަން އިލާހާ ކުއްތަން ނުވެއެވެ. އަދި ތިމަން އިލާހުގެ އަޅާ ދެކެ ލޯބިވެ ވޮޑިގެންފުމަށް ދާންދެން، އެ އަޅާ ސުންނަތް އަޅުކަންތައް އަދާކޮށް ތިމަން އިލާހަށް ކުއްތަންވުން މަތީ ދެމިހުރެއެވެ. އަދި އެ އަޅާ ދެކެ ތިމަން އިލާހު ލޯބިވެ ވޮޑިގެންފި ނަމަ، އެ އަޅާގެ އަޑު އަހައި އުޅޭ އިވުން ކަމުގައި، އަދި ބަލާ އުޅޭ ބެލުން ކަމުގައި، އަދި ހިފާ އުޅޭ އަތް ކަމުގައި، އަދި ހިނގައި އުޅޭ ފައި ކަމުގައި ތިމަން އިލާހު ވެވޮޑިގަންނަވާނަމެވެ. (އެބަހީ: ﷲ، އެ އަޅާގެ ޢަމަލުތައް އިސްލާހުކޮށް ސީދާ ކުރައްވާނެއެވެ. އަދި އެ ގުނަވަންތަކުން އޭނާ ކޮށްއުޅޭނީ މުޅިން ވެސް ﷲ ރުހިވޮޑިގަންނަވާ ޢަމަލުތަކެވެ.) އަދި އެ އަޅާ ކަމަކަށް އެދި ތިމަން އިލާހަށް ދަންނަވައިފި ނަމަ، ތިމަން އިލާހު އެ ކަމެއް އެ އަޅާއަށް ދެއްވާނެކަށް ކަށަވަރެވެ. އަދި އެ އަޅާ ރައްކާތެރިކޮށް ދެއްވުމަށް އެދި ތިމަން އިލާހަށް ދަންނަވައިފި ނަމަ، ތިމަން އިލާހު އެ އަޅާއަށް ރައްކާތެރިކަން ދެއްވާނެކަށް ކަށަވަރެވެ.“

[يَا ابْنَ آدَمَ، إِنَّكَ مَا دَعَوتَنِي وَرَجَوتَنِي]

[އޭ އާދަމުގެ ދަރިއާއޭ، ތިމަން އިލާހަށް ދުޢާކޮށް ދަންނަވާހައި ހިނދަކު]

عَنْ أَنَسٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «قَالَ اللَّهُ تَعَالَى: يَا ابْنَ آدَمَ! إِنَّكَ مَا دَعَوتَنِي وَرَجَوتَنِي؛ غَفَرْتُ لَكَ عَلَى مَا كَانَ مِنْكَ وَلاَ أُبَالِي. يَا ابْنَ آدَمَ! لَو بَلَغَتْ ذُنُوبُكَ عَنَانَ السَّمَاءِ، ثُمَّ اسْتَغْفَرْتَنِي؛ غَفَرْتُ لَكَ. يَا ابْنَ آدَمَ! إِنَّكَ لَو أَتَيتَنِي بِقُرَابِ الْأَرْضِ خَطَايَا، ثُمَّ لَقِيتَنِي لَا تُشْرِكُ بِي شَيئًا؛ لَأَتَيتُكَ بِقُرَابِهَا مَغْفِرَةً»

އަނަސް رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: (ޤުދުސީ ޙަދީޘެއްގައި) ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ: ”ﷲ ތަޢާލާ ވަޙީ ކުރެއްވިއެވެ: އޭ އާދަމުގެ ދަރިއާއެވެ. ހަމަކަށަވަރުން ތިބާ، ތިމަން އިލާހަށް ދުޢާކޮށް ދަންނަވައި އަދި ތިމަން އިލާހުގެ ޙަޟްރަތުން އުންމީދު ކުރާހާ ހިނދަކު، ތިބާގެ ކިބައިން ވެވިފައިވާ ފާފަތައް ތިމަން އިލާހު ފުއްސަވައި ދެއްވާނަމެވެ. އަދި (އެ ފާފަތަކަށް) ތިމަން އިލާހު ނުބައްލަވާނަމެވެ. އޭ އާދަމުގެ ދަރިއާއެވެ. އުޑުގެ ވިލާތަކާ ހަމައަށް ތިބާގެ ފާފަތައް ފޯރިކަމުގައި ވިޔަސް، ފާފަ ފުއްސަވައި ދެއްވުމަށް އެދި ތިބާ ދަންނަވައިފި ނަމަ، ތިމަން އިލާހު ތިބާއަށް ފާފަ ފުއްސަވައި ދެއްވާނަމެވެ. އޭ އާދަމުގެ ދަރިއާއެވެ. ހަމަކަށަވަރުން ބިން ފުރޭހައި ވަރަށް ތިބާ ފާފަތަކައިގެން ތިމަން އިލާހާ ބައްދަލުވި ކަމުގައި ވިޔަސް، ތިބާ ތިމަން އިލާހާ އެއްވެސް ފަރާތެއް ޝަރީކުކޮށްފައި ނުވާ ނަމަ، ހަމަ އެއާ އެއްވަރަށް (އެބަހީ: ބިން ފުރޭހައި ވަރަށް) ތިމަން އިލާހު ތިބާއާ ފާފަ ފުއްސެވުމައިގެން ތިމަން އިލާހު ތިބާއާ ބައްދަލު ކުރައްވާނެކަން ކަށަވަރެވެ.“

#45

[إِنَّ اللَّهَ حَرَّمَ عَلَيهِمُ الشُّحُومَ فَأَكَلُوا ثَمَنَهُ]

[އެބައިމީހުންނަށް ސަރުބީ ޙަރާމް ކުރެއްވުމުން، އެ ވިއްކައިގެން އޭގެ އަގު ކައި އުޅުނެވެ]

عَنْ جَابِرٍ رَضِيَ اللَّهُ عَنْهُ أَنَّهُ سَمِعَ رَسُولَ اللَّهِ ﷺ عَامَ الْفَتْحِ -وَهُوَ بِمَكَّةَ- يَقُولُ: «إِنَّ اللَّهَ وَرَسُولَهُ حَرَّمَ بَيعَ الْخَمْرِ، وَالْمَيتَةِ، وَالْخِنْزِيرِ، وَالْأَصْنَامِ» فَقِيلَ: يَا رَسُولَ اللَّهِ! أَرَأَيتَ شُحُومَ الْمَيتَةِ، فَإِنَّهُ يُطْلَى بِهَا السُّفُنُ، وَيُدْهَنُ بِهَا الْجُلُودُ، وَيَسْتَصْبِحُ بِهَا النَّاسُ؟ قَالَ: «لَا؛ هُوَ حَرَامٌ» ثُمَّ قَالَ رَسُولُ اللَّهِ ﷺ عِنْدَ ذَلِكَ: «قَاتَلَ اللَّهُ الْيَهُودَ؛ إِنَّ اللَّهَ حَرَّمَ عَلَيهِمُ الشُّحُومَ، فَأَجْمَلُوهُ، ثُمَّ بَاعُوهُ، فَأَكَلُوا ثَمَنَهُ»

ޖާބިރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: މައްކާ ފަތަޙަ ކުރެއްވި އަހަރު، ރަސޫލު ﷲ ﷺ އެ ބިމުގައި ހުންނަވައިގެން ޙަދީޘް ކުރައްވަނިކޮށް ތިމަން އަޑު އެހީމެވެ : ”ހަމަކަށަވަރުން ﷲ އާއި އެ އިލާހުގެ ރަސޫލާ ވަނީ ރަލާއި، މުޅަ ތަކެއްޗާއި، އޫރު މަހާއި، ބުދުތަކުގެ ވިޔަފާރިކުރުން ޙަރާމް ކުރައްވާފައެވެ.“ ދެން ނަބިއްޔާއަށް ދެންނެވުނެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. މުޅަ ތަކެތީގެ ސަރުބީއާމެދު ދެކިލައްވަނީ ކިހިނަކުންތޯއެވެ؟ އެހެނީ ހަމަކަށަވަރުން އޮޑިދޯނިފަހަރުގައި ( ތެޔޮ ހާކައިގެން ފެން ވަނަ ނުދިނުމަށް) ކަޑިތާރުލާން، އެ މުޅަ ތަކެތީގެ ސަރުބީގެ ތެޔޮ ބޭނުން ކުރެވެއެވެ. އަދި ދިބާޣުކުޅަހަންތަކުގައި ތެޔޮ ހޭކުމަށް ވެސް އޭގެ ބޭނުން ކުރެވެއެވެ. އަދި ވޮށް ދިއްލައި އަލި ކުރުމަށް ވެސް މީސްތަކުން އޭގެ ބޭނުން ކުރެއެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ނޫނެކެވެ. އެ ހަރާމްވެގެން ވެއެވެ.“ އެއަށްފަހު ރަސޫލު ﷲ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ ޔަހޫދީން ޤަތުލު ކުރައްވާށިއެވެ. (އެބަހީ: އެބައިމީހުންނަށް ލަޢުނަތް ލައްވާށިއެވެ.) ހަމަކަށަވަރުން ﷲ އެބައިމީހުންނަށް ސަރުބީ ޙަރާމް ކުރެއްވުމުން، އެބައިމީހުން އެ ސަރުބީ ވިރުވައި ތެލަށް ހަދައިގެން، ދެން އެ ވިއްކައިގެން އޭގެ އަގު ކައި އުޅުނެވެ.“ (އެބަހީ: އޭގެން ލިބުނު ފައިސާ ބޭނުން ކުރިއެވެ.)

[كُلُّ مُسْكِرٍ حَرَامٌ]

[މަސްތުވާ ކޮންމެ އެއްޗެއް ޙަރާމްވެގެން ވެއެވެ]

عَنْ أَبِي بُرْدَةَ، عَنْ أَبِيهِ -أَبِي مُوسَى الْأَشْعَرِيِّ- رَضِيَ اللَّهُ عَنْهُ: أَنَّ النَّبِيَّ ﷺ بَعَثَهُ إِلَى الْيَمَنِ، فَسَأَلَهُ عَنْ أَشْرِبَةٍ تُصْنَعُ بِهَا. فَقَالَ: «وَمَا هِيَ؟» قَالَ: الْبِتْعُ وَالْمِزْرُ. -فَقِيلَ لِأَبِي بُرْدَةَ: وَمَا الْبِتْعُ؟ قَالَ: نَبِيذُ الْعَسَلِ، وَالْمِزْرُ نَبِيذُ الشَّعِيرِ- فَقَالَ: «كُلُّ مُسْكِرٍ حَرَامٌ» خَرَّجَهُ الْبُخَارِيُّ.
وَخَرَّجَهُ مُسْلِمٌ وَلَفْظُهُ: قَالَ: بَعَثَنِي رَسُولُ اللَّهِ ﷺ أَنَا وَمُعَاذٌ إِلَى الْيَمَنِ، فَقُلْتُ: يَا رَسُولَ اللَّهِ! إِنَّ شَرَابًا يُصْنَعُ بِأَرْضِنَا يُقَالُ لَهُ: الْمِزْرُ مِنَ الشَّعِيرِ، وَشَرَابٌ يُقَالُ لَهُ: الْبِتْعُ مِنَ الْعَسَلِ، فَقَالَ: «كُلُّ مُسْكِرٍ حَرَامٌ»
وَفِي رِوَايَةٍ لِمُسْلِمٍ: فَقَالَ: «كُلُّ مَا أَسْكَرَ عَنِ الصَّلَاةِ فَهُوَ حَرَامٌ»
وَفِي رِوَايَةٍ لَهُ: وَكَانَ رَسُولُ اللَّهِ ﷺ قَدْ أُعْطِي جَوَامِعَ الْكَلِمِ بِخَوَاتِمِهِ، فَقَالَ: «أَنْهَى عَنْ كُلِّ مُسْكِرٍ أَسْكَرَ عَنِ الصَّلَاةِ»

އަބޫ ބުރްދާ، އެ ޞަޙާބީގެ ބައްޕާފުޅު އަބޫ މޫސާ އަލްއަޝްޢަރީ رَضِيَ اللَّهُ عَنْهُ ގެ އަރިހުން ރިވާ ކުރެއްވިއެވެ: ހަމަކަށަވަރުން ނަބިއްޔާ ﷺ، އަބޫ މޫސާ އަލްއަޝްޢަރީ ޔަމަނަށް ފޮނުއްވިއެވެ. ފަހެ އެ ރަށުގައި އުފައްދައި އުޅުނު ބުއިންތަކަކާ ބެހޭ ގޮތުން އަބޫ މޫސާ އަލްއަޝްޢަރީ، ރަސޫލާގެ އަރިހު ސުވާލު ކުރެއްވިއެވެ. ރަސޫލާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެއީ ކޮން ބުއިންތަކެއްތޯއެވެ؟“ އަބޫ މޫސާ އަލްއަޝްޢަރީ ދެންނެވިއެވެ: ބިތުޢުއާއި މިޒްރުއެވެ. -(ޙަދީޘްގެ ރާވީކަމަށްވާ) އަބޫ ބުރްދާއަށް ދެންނެވުނެވެ: ބިތުޢުއާއި މިޒްރުއަކީ ކޮން އެއްޗެއްސެއްތޯއެވެ؟ އަބޫ ބުރްދާ ދެންނެވިއެވެ: ބިތުޢުއަކީ ފެނަށް މާމުއި އަޅައިގެން ހަދާ ރަލެކެވެ. އަދި މިޒްރުއަކީ ފެނަށް ހިމަގޮދަން އަޅައިގެން ހަދާ ރަލެކެވެ- ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މަސްތުވާ ކޮންމެ އެއްޗެއް ޙަރާމްވެގެން ވެއެވެ.“
އަދި މުސްލިމުގެ ރިވާޔަތުގައި ވެއެވެ: އަބޫ މޫސާ އަލްއަޝްޢަރީ ވިދާޅުވިއެވެ: ރަސޫލު ﷲ ﷺ، ތިމަންނައާއި މުޢާޛު ޔަމަނަށް ފޮނުއްވިއެވެ. ތިމަން ދެންނެވީމެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. ހަމަކަށަވަރުން ތިމަންމެންގެ ރަށުގައި ހިމަ ގޮދަނުން ހަދާ އުޅޭ، މިޒްރު ކިޔާ ބުއިމެއް ވެއެވެ. އަދި މާމުއިން ހަދާ ބިތުޢު ކިޔާ ބުއިމެއް ވެސް ވެއެވެ. ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މަސްތުވާ ކޮންމެ އެއްޗެއް ޙަރާމްވެގެން ވެއެވެ.“
އަދި މުސްލިމުގެ އެހެން ރިވާޔަތެއްގައި ވެއެވެ: ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މަސްތު ކުރުވައިގެން ނަމާދަށް ހުރަސް އަޅާ ކޮންމެ އެއްޗެއް ޙަރާމްވެގެން ވެއެވެ.“
އަދި މުސްލިމުގެ އެހެން ރިވާޔަތެއްގައި ވެއެވެ: އަދި ހަމަކަށަވަރުން ރަސޫލު ﷲ ﷺ އަށް ނިންމަވަނިވި ޖަވާމިޢުލް ކަލިމި ދެއްވުނެވެ. (ޖަވާމިޢުލް ކަލިމިއަކީ: އިބާރާތްތައް ކުރު، އެކަމަކު މާނަ ފުން ބަސްފުޅުތަކެވެ.) ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”މަސްތު ކުރުވައިގެން ނަމާދަށް ހުރަސް އަޅާ ކޮންމެ އެއްޗެއް، ތިމަން ނަބިއްޔާ މަނާ ކުރީމެވެ.“

[مَا مَلَأَ آدَمِيٌّ وِعَاءً شَرًّا مِنْ بَطْنٍ]

[އާދަމުގެ ދަރިޔާ ބަނޑަށް ވުރެ ނުބައި ކަންވާރެއް ނުފުރައެވެ]

عَنِ الْمِقْدَامِ بْنِ مَعْدِي كَرِبَ رَضِيَ اللَّهُ عَنْهُ قَالَ: سَمِعْتُ رَسُولَ اللَّهِ ﷺ يَقُولُ: «مَا مَلَأَ آدَمِيٌّ وِعَاءً شَرًّا مِنْ بَطْنٍ، بِحَسْبِ ابْنِ آدَمَ أَكَلَاتٌ يُقِمْنَ صُلْبَهُ، فَإِنْ كَانَ لَا مَحَالَةَ، فَثُلُثٌ لِطَعَامِهِ، وَثُلُثٌ لِشَرَابِهِ، وَثُلُثٌ لِنَفَسِهِ»

عَنْ عَبْدِ اللَّهِ بْنِ عَمْرٍو رَضِيَ اللَّهُ عَنْهُمَا عَنِ النَّبِيِّ ﷺ قَالَ: «أَرْبَعٌ مَنْ كُنَّ فِيهِ كَانَ مُنَافِقًا، وَإِنْ كَانَتْ خَصْلَةٌ مِنْهُنَّ فِيهِ كَانَتْ فِيهِ خَصْلَةٌ مِنَ النِّفَاقِ حَتَّى يَدَعَهَا: مَنْ إِذَا حَدَّثَ كَذَبَ، وَإِذَا وَعَدَ أَخْلَفَ، وَإِذَا خَاصَمَ فَجَرَ، وَإِذَا عَاهَدَ غَدَرَ»

[لَو أَنَّكُمْ كُنْتُمْ تَوَكَّلُونَ عَلَى اللَّهِ حَقَّ تَوَكُّلِهِ]

[ﷲ އާ ވަކީލު ކުރަން ޖެހޭ ޙައްޤު ގޮތުގައި ތިޔަބައިމީހުން އެ އިލާހާ ވަކީލު ކުރާ ނަމަ]

عَنْ عُمَرَ بْنِ الْخَطَّابِ رَضِيَ اللَّهُ عَنْهُ عَنِ النَّبِيِّ ﷺ قَالَ: «لَو أَنَّكُمْ كُنْتُمْ تَوَكَّلُونَ عَلَى اللَّهِ حَقَّ تَوَكُّلِهِ، لَرَزَقَكُمْ كَمَا يَرْزُقُ الطَّيرَ، تَغْدُو خِمَاصًا، وَتَرُوحُ بِطَانًا»

ޢުމަރު ބިން އަލްޚައްޠާބު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ނަބިއްޔާ ﷺ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ އާ ވަކީލު ކުރަން ޖެހޭ ޙައްޤު ގޮތުގައި ތިޔަބައިމީހުން އެ އިލާހާ ވަކީލު ކުރާ ނަމަ، ﷲ ދޫނިތަކަށް ރިޒުޤު ދެއްވާ ފަދައިން އެ އިލާހު ތިޔަބައިމީހުންނަށް ވެސް ރިޒުޤު ދެއްވާނެކަން ކަށަވަރެވެ. ހެނދުނު ހެނދުނާ އެ ދޫނިތައް ނުކުމެގެން ދަނީ ހުސްބަނޑާއެވެ. އަދި ހަވީރު އެނބުރި އަންނަނީ ބަނޑު ފުރައިގެންނެވެ.“

رَوَاهُ الْإِمَامُ أَحْمَدُ [00205]، وَالتِّرْمِذِيُّ [2344]، وَالنَّسَائِيُّ [فِي الْكُبْرَى 11805]، وَابْنُ مَاجَهْ [4164]، وَابْنُ حِبَّانَ فِي صَحِيحِهِ [0730]، وَالْحَاكِمُ [7894]، وَقَالَ التِّرْمِذِيُّ: ”حَسَنٌ صَحِيحٌ“ [وَصَحَّحَهُ الأَلبَانِي فِي صَحِيح سُنَن التِّرمِذِي]

މުސްނަދު އަޙުމަދު 00205، ތިރްމިޛީ 2344 (ޙަސަން)، ނަސާއީގެ ސުނަން އަލްކުބްރާ 11805، އިބްނު މާޖާ 4164، އިބްނު ޙިއްބާން 0730، އަލްޙާކިމް 7894

[لَا يَزَالُ لِسَانُكَ رَطْبًا مِنْ ذِكْرِ اللَّهِ]

[ﷲގެ ޒިކުރުކުރުން މަތީ ތިބާގެ ދޫ ހިކުނަ ނުދޭށެވެ]

عَنْ عَبْدِ اللَّهِ بْنِ بُسْرٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: أَتَى النَّبِيَّ ﷺ رَجُلٌ، فَقَالَ: يَا رَسُولَ اللَّهِ! إِنَّ شَرَائِعَ الْإِسْلَامِ قَدْ كَثُرَتْ عَلَينَا، فَبَابٌ نَتَمَسَّكُ بِهِ جَامِعٌ؟ قَالَ: «لَا يَزَالُ لِسَانُكَ رَطْبًا مِنْ ذِكْرِ اللَّهِ عَزَّ وَجَلَّ» خَرَّجَهُ الْإِمَامُ أَحْمَدُ بِهَذَا الَّفْظِ.
وَخَرَّجَهُ التِّرْمِذِيُّ، وَابْنُ مَاجَهْ، وَابْنُ حِبَّانَ فِي صَحِيحِهِ بِمَعْنَاهُ، وَقَالَ التِّرْمِذِيُّ: ’حَسَنٌ غَرِيبٌ‘.
وَكُلُّهُمْ خَرَّجَهُ مِنْ رِوَايَةِ عَمْرِو بْنِ قَيسٍ الْكِنْدِيِّ، عَنْ عَبْدِ اللَّهِ بْنِ بُسْرٍ رَضِيَ اللَّهُ عَنْهُ.
وَخَرَّجَ ابْنُ حِبَّانَ فِي صَحِيحِهِ وَغَيرُهُ مِنْ حَدِيثِ مُعَاذِ بْنِ جَبَلٍ رَضِيَ اللَّهُ عَنْهُ قَالَ: آخِرُ مَا فَارَقْتُ عَلَيهِ رَسُولَ اللَّهِ ﷺ أَنْ قُلْتُ لَهُ: أَيُّ الْأَعْمَالِ خَيرٌ وَأَقْرَبُ إِلَى اللَّهِ؟ قَالَ: «أَنْ تَمُوتَ وَلِسَانُكَ رَطْبٌ مِنْ ذِكْرِ اللَّهِ عَزَّ وَجَلَّ»

ޢަބްދު ﷲ ބިން ބުސްރު رَضِيَ اللَّهُ عَنْهُ ރިވާ ކުރެއްވިއެވެ: ނަބިއްޔާ ﷺ ގެ އަރިހަށް މީހަކު އައިސް ދެންނެވިއެވެ: އޭ ﷲގެ ރަސޫލާއެވެ. ހަމަކަށަވަރުން އިސްލާމް ދީނުގެ ޝަރާއިޢުތައް (އެބަހީ: ދީނުގައި ކުރުން ހެޔޮ އިތުރު ވެގެންވާ އަޅުކަންތައްތައް) ތިމަންމެންނަށް ނުހަނު ގިނަ ވެއްޖެއެވެ. ވީމާ ތިމަންމެން އެ ކަމެއްގައި ހިފަހައްޓާނެ ޖާމިޢު (އެބަހީ: ފަސޭހަ އަދި ގިނަ ސަވާބު ލިބޭނެ) ކަމެއް (ވިދާޅުވެދެއްވާށެވެ.) ‎ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”ﷲ ޢައްޒަ ވަޖައްލަގެ ޒިކުރުކުރުން މަތީ ތިބާގެ ދޫ ހިކުނަ ނުދޭށެވެ.“ (އެބަހީ: އަބަދުވެސް ﷲގެ ޒިކުރުކުރުން މަތީ ދެމިހުންނާށެވެ.)
އިމާމު އަޙުމަދު މި ޙަދީޘް ރިވާ ކުރެއްވީ މި ލަފުޒުންނެވެ. އަދި މީގެ އިތުރުން މި ޙަދީޘް ރިވާ ކުރެއްވީ ތިރްމިޛީއާއި އިބްނު މާޖާއާ، އިބްނު ޙިއްބާނު އެކަލޭގެފާނުގެ ޞަޙީޙުގައި މި މާނައިގައެވެ. އަދި ތިރްމިޛީ ވިދާޅުވިއެވެ: ’މިއީ ޙަސަން ޣަރީބު ޙަދީޘެކެވެ.‘
އަދި އެ އެންމެން ވެސް މި ޙަދީޘް ރިވާ ކުރެއްވީ، ޢަމްރު ބިން ޤައިސް އަލްކިންދީގެ ރިވާޔަތުން، ޢަބްދު ﷲ ބިން ބުސްރި رَضِيَ اللَّهُ عَنْهُ ގެ އަރިހުންނެވެ.
އަދި އިބްނު ޙިއްބާނުގެ ޞަޙީޙުގައި އެކަލޭގެފާނު އަދި އެ ނޫން ވެސް ބޭކަލުން، މުޢާޛު ބިން ޖަބަލު رَضِيَ اللَّهُ عَنْهُ ގެ ޙަދީޘް ރިވާ ކުރެއްވިއެވެ. މުޢާޛު ވިދާޅުވިއެވެ: ރަސޫލު ﷲ ﷺ އާ އެންމެ ފަހު ފަހަރު ތިމަން ވަކިވި އިރު، ރަސޫލާ އަރިހު ދެންނެވީމެވެ: ﷲގެ ޙަޟްރަތުގައި އެންމެ ހެޔޮވެގެންވާ އަދި އެންމެ ކުއްތަންވެގެންވާ ޢަމަލީ ކޮބައިތޯއެވެ؟ ނަބިއްޔާ ޙަދީޘް ކުރެއްވިއެވެ: ”އެއީ ތިބާ މަރުވެގެން ދާއިރު، ﷲ ޢައްޒަ ވަޖައްލަގެ ޒިކުރު ކުރުމައިގެން ތިބާގެ ދޫ ތެތްކޮށް ހުރުމެވެ.“ (އެބަހީ: ﷲގެ ޒިކުރުކުރުން މަތީ ތިބާ މަރުވެގެން ދިއުމެވެ.)

خَرَّجَهُ الْإِمَامُ أَحْمَدُ [17680] بِهَذَا الَّفْظِ. وَخَرَّجَهُ التِّرْمِذِيُّ [3375]، وَابْنُ مَاجَهْ [3793]، وَابْنُ حِبَّانَ فِي صَحِيحِهِ [0814] بِمَعْنَاهُ، وَقَالَ التِّرْمِذِيُّ: ”حَسَنٌ غَرِيبٌ“ [وَصَحَّحَهُ الأَلبَانِي فِي صَحِيح سُنَن التِّرمِذِي]
وَخَرَّجَ ابْنُ حِبَّانَ فِي صَحِيحِهِ [0818، وَصَحَّحَهُ الأَلبَانِي فِي التَّعلِيقَات الحِسَان عَلَى صَحِيح ابن حِبَّان 815] وَغَيرُهُ.
