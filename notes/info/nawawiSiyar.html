بِسمِ اللهِ الرَّحمَنِ الرَّحِيمِ

ޙަދީޘްއެމްވީ - ޛަހަބީގެ ސިޔަރު އަޢުލާމް އައްނުބަލާއިން ނަވަވީގެ ވަނަވަރު



ތިރީގައި ވަނީ އައްނަވަވީއާމެދު، ތާރީޚުގެ މަޝްހޫރު އިލްމުވެރިއެއް ކަމަށްވާ ޝަމްސުއްދީން އައްޛަހަބީ (748-673 ހ.) ލިޔުއްވި ވަނަވަރުން ނަގާފައިވާ މައުލޫމާތެވެ. [سِيَرُ أَعلَام النُّبَلَاء لِلذَّهَبِي، الجُزءُ المَفقُودُ، تَرجَمَةُ النوَوِي (6445) ص 340]

النَّوَاوِيُّ، الشَّيخُ الإِمَامُ القُدوَةُ الحَافِظُ الزَّاهِدُ العَابِدُ الفَقِيهُ المُجتَهِدُ الرَّبَانِيُّ شَيخُ الإِسلَامِ أَحسِبُهُ. الإِمَامُ مُحيِي الدِّينُ أَبُو زَكَرِيَّا يَحيَى بنُ شَرَفِ بنِ مُرِّيِّ بنِ حَسَنِ بنِ حُسَينِ بنِ مُحَمَّدٍ بنِ حِزَامِ الحِزَامِيِّ الحَورَانِيِّ النَّوَاوِي الشَّافِعِي.

صَاحِبُ التَّصَانِيفَ الَّتِي سَارَت بِهَا الرُّكبَانَ، وَاشتَهَرَت بِأَقَاصِى البُلدَانِ.

وُلِدَ فِي المُحَرَّمِ سَنَةِ إِحدَى وَثَلَاثِينَ وَسِتُّمِائَة بِنَوَى، وَكَانَ أَبُوهُ دُكَّانِيًّا بِهَا، فَنَشَأَ الشَّيخُ فِي سِترٍ وَخَيرٍ، وَحَفِظَ القُرآنَ، وَبَقِيَ يَتَعَيَّش فِي الدُّكَّانِ لِأَبِيهِ، ثُمَّ نَقَلَهُ أَبُوهُ فِي سَنَةِ تِسعٍ وَأَربَعِينَ إِلَى دِمِشقَ لِيَشتَغِلَ بِهَا، فَنَزَلَ بِالرَّوَاقِيَةِ يَتَقَوَّتُ بِالجرَايَةِ، وَيَدرُسُ فِي ’التَّنبِيهِ‘ فَحَفِظَهُ فِي أَربَعَةِ أَشهُرٍ وَنِصفٍ، وَقَرَأَ رُبعَ ’المُهَذَّب‘ فِي تَمَام السَّنَة، عَلَى الشَّيخِ الكَمَالِ إِسحَاقُ بِن أَحمَدَ.

ثُمَّ حَجَّ مَعَ وَالِدِهِ، وَقَد لَاحَت عَلَيهِ أَمَارَات النَّجَابَةِ وَالفَهمِ، فَاتَّفَقَ أَنَّهُ أَقَامَ بِالمَدِينَةِ النَّبَوِيَّةِ شَهرًا وَنِصفًا، وَتَعَلَّلَ فِي أَكثَرِ الطَّرِيقِ، وَرَجَعَ وَأَكَبَّ عَلَى طَلَبِ العِلمِ لَيلًا وَنَهَارًا اشتِغَالًا، فَضُرِبَ بِهِ المَثَل، وَهَجَرَ النَّومَ إِلَّا عَن غَلَبَةٍ، وَضَبَطَ أَوقَاتُهُ إِلَّا بِلُزُومِ الدَّرسِ أَو الكِتَابَةِ أَو المُطَالَعَةِ، أَو التَّرَدُّدِ إِلَى الشُّيُوخِ، وَتَرَكَ كُلّ رَفَاهِيَة وَتَنَعُّم، مَعَ تَقوَى وَقَنَاعَةِ وَوَرَعِ وَحُسنِ مُرَاقَبَةِ لِلَّه فِي السِرِّ وَالعَلَانِيَّةِ، وَتَرَكَ رُعُونَاتِ النَّفسِ، مِن ثِيَابٍ حَسَنَة، وَمَآكِل طَيِّبَة، وَتَجَمّل هَيئَة، بَل طَعَامُهُ جَلَفُ الخُبز يَابِسَة، وَلِبَاسُهُ خَام، وَشَيخَانِيتُهُ لَطِيفَة، فَرَحِمَهُ اللَّهُ وَرَضِيَ عَنهُ وَجَزَاهُ عَن العِلمِ خَيرًا.

ذَكَرَ صَاحِبُهُ الشَّيخُ أَبُو الحَسَن عَلِي ابن العَطَّر: أَنَّ الشَّيخ مُحيِي الدِّين حَّدَثَهُ أَنَّهُ كَانَ يَقرَأُ كُلّ يَوم اثنَى عَشَرَ دَرسًا عَلَى مَشَايخِهِ، شَرحًا وَتَصحِيحًا، دَرسَينِ فِي «الوَسِيطِ»، وَدَرسًا فِي «المُهَذَّبِ»، وَدَرسًا فِي «الجَمعِ بيَنَ الصَّحِيحَينِ»، وَدَرسًا فِي «صَحِیحِ مُسلِمٍ»، وَدَرسًا فِي «اللُّمَعِ» لِابنِ جِنِّي، وَدَرسًا فِي التَّصرِيفِ، وَدَرسًا فِي أُصُولِ الفِقهِ، وَدَرسًا فِي أَسمَاءِ الرِّجَالِ، وَدَرسًا فِي أُصُولِ الدِّينِ.

قَالَ: وَكُنتُ أُعَلّق جَمِيع مَا يَتَعَلَّق بِهَا، مِن شَرح مُشكِلٍ، وَوُضُوح عِبَارَة، وَضَبطِ لُغَة، وَبَارَكَ اللهُ لِي فِي وَقتِي، وَخَطَرَ لِي أَن أَشتَغِلَ بِالطِّب وَاشتَرَيتُ كِتَابَ «القَانُونَ»، فَأظلَمَ قَلبِي، وَبَقِيتُ أَيَّامًا لَا أَقدر عَلَى الاشتِغَال، فَأَفقتُ عَلَى نَفسِي، وَبِعتُ القَانُونَ فَأَنَارَ قَلبِي.

وَلَاَزمَ الاشتِغَال وَالتَّصنِيف وَالإِفَادَة، مُحتَسِبًا فِي ذَلِكَ، مُبتَغِيًا وَجهَ اللهِ، وَالصَوم وَالتَّهَجُّد وَالذِّكر وَالأَورَاد، وَحِفظُ الجَوَارِح، وَذَمّ النَّفس، وَصَبَرَ عَلَى العيَش الخَشن، مُلَاَزمَةً كُلِّيَّة، لَا مَزِيد عَلَيهَا.

قَالَ ابنُ العَطَّر: ذَكَرَ لِي شَيخُنَا أنَّهُ كَانَ لَا يضِيع لَهُ وَقتًا فِي لَيلٍ وَلَا نَهَار إِلَّا فِي اشتِغَالٍ، حَتَّى فِي الطُّرُقِ، وَأَنَّهُ دَامَ عَلَى هَذَا سِتّ سِنِين، ثُمَّ أَخَذَ فِي التَّصنِيفِ وَالإِفَادَةِ وَالنَّصِيحَةِ، وَقَولِ الحَقِّ.

قُلتُ: كَانَ مَعَ مُلَازمَتِهِ التَّامَّة لِلعِلمِ وَمُوَاظَبَتِهِ لَهُ، فَائِق الوَرع، وَتَزكِيَة النَّفس مِن شَوَائِب الهَوَى، وَسَيِّئ الأَخلَاق، وَمَحَقَهَا مِن أَغرَاضِهَا، عَارِفًا بِالحَدِيثِ ، قَائِمًا عَلَى أَكثَرِ فُنُونِهِ، عَارِفًا بِرِجَالِهِ، رَأسًا فِي نَقلِ المَذهَبِ، مُتَضَلِّعًا فِي عُلُومِ الإِسلَامِ.

قَالَ شَيخُنَا الرَّشِيدُ الحَنَفِي ابنُ المُعَلِّمِ: عَذَلتُ الشَّيخَ مُحيِي الدِّين فِي تَركِهِ الحَمَّام، وَضَيق العَيش، وَخَوَّفتُهُ مِن مَرضٍ يُعَطِّلُهُ عَنِ العِلمِ، فَقَالَ: إِنَّ فُلَانًا صَامَ حَتَّى اخضَر جِلده.

كَانَ الشَّيخُ يَمتَنِع جُملَة مِن أَكلِ الخِيَارِ وَالفَاكِهَة، وَيَقُولُ: أَخَافُ تُرَطِّبُنِي وَتَجلِب النَّوم، وَكَانَ يَأَكُل فِي اليَومِ وَاللَيلَة غَالِبًا أَكلَةً وَاحِدَةً، ثُمَّ يَشرَب مَرَّةً عِندَ السَّحَر.

قَالَ ابنُ العَطَّار: كَلَّمتُهُ فِي الفَاكِهَة، فَقَالَ: دِمَشق كَثِيرَة الأَوقَاف، وَأَملَاك المَحجُورِ عَلَيهِم، ثُمَّ المُعَامَلَة فِيهَا عَلَى وَجهِ المُسَاقَاة، وَفِيهَا خِلَاف، فَكَيفَ تطِيب نَفسِي بِأَكلِ ذَلِك؟

وَكَانَ لَا يَقبل مِن أَحَدٍ شَيئًا إِلَّا ِفي النَّادِر، يَقبَلُ شَيئًا يَسِيرًا مِمَّن لَا يَشتَغِل عَلَيهِ، قَد أَهدَى لَهُ فَقِيرٌ إِبرِيقًا فَقَبِلَهُ، وَعَزمَ عَلَيهِ صَاحِبُهُ الخَطِيبُ بُرهَانُ الدِّينِ الِإسَكَندَرَانِي أَن يُفطِر مَعَهُ، قَالَ: هَاتِ الطَّعَام وَنفطر مَعًا، فَأَكَلَ مِنهُ وَكَانَ لَونَين، وَقَلَّ أَن كَانَ يَأكُل إِدَامَين، وَكَانَ قَلِيل الضَّحك، عَدِيم اللَّعِب، بَل هُوَ جِد صَرف، يَقُول الحَقّ، وَإِن كَانَ عَلَيهِ، لَا تَأخُذهُ فِي اللهِ لَومَة لَائِمٍ، وَيُوَاجِه الأُمَرَاء وَالظُّلم بِالإِنكَارِ، وَيَكتُب إِليَهِم، وَيُخَوِّفُهُم بِاللهِ.

قَالَ ابنُ فَرح - وَكَانَ مِمَّن يَشرَحُ عَلَى الشَّيخِ - صَارَ الشَّيخ مُحيِي الدِّين إِلَى ذَلِك رَتب لَو نَهَضَ رَجُلٌ مِنهَا لَشدَّت إِلَيهِ الرِّحَال: العِلم وَالزُّهد وَالأَمر بِالمَعرُوفِ وَالنَّهي عَنِ المُنكَرِ.

وَكَانَ الشَّيخ - رَضِيَ اللهُ عَنْهُ - يَقتَنِع بِاليَسِيرِ، وَوَلِيَ مَشيَخَةَ دَارِ الحَدِيثِ الأَشرَفِيَّةِ مَعَ صِغَرِ سِنِّهِ، وَنُزُول رِوَايَتِهِ فِي حَيَاةِ مَشَايخِهِ بَعدَ الإمَام أَبِي شَامَة، فَمَا أَجِد مَا مَكَّنَهُ فِيمَا بَلَغَنِي، بَل كَانَ يَجِيئه مِن وَالِدِهِ شَيء يقتَات مِنهُ، وَاشتَرَى بِالجَامِكِيَّةِ كُتُبًا وَفِقهًا، سَافَرَ وَزَارَ بَیت المَقدِس، فَردَّ إِلَى نَوَى مَرِيضًا، وَالنتقلَ بِهِ إِلَى اللهِ فِي الرَّابِع وَالعِشرِين مِن رَجَب سَنَةَ سِتّ وَسَبعِين وَسِتّمِائَة.

قَالَ قُطُب الدِّین مُوسَى شَيخُنَا: كَانَ أَوحَد زَمَانِهِ فِي العِلمِ وَالزُّهدِ وَالوَرعِ وَالعِبَادَةِ وَالتَقلِيلِ وَخُشُونَة العَيش، وَاقَفَ المَلِك الظَّاهِرُ بدَار العَدل غَير مَرَّة، فَحُكِيَ عَنهُ قَالَ: أَنَا أُفَزّعُ مِن هَذَا، وَقَالَ الفَقِيهُ شَمسُ الدِّين مُحَمَّدُ بنِ الفَخر: كَانَ إِمَامًا بَارِعًا حَافِظًا مُفتِيًا، أَتقَنَ عُلُومًا شَتًى، وَصَنَّفَ بِالتَصَانِيفِ الحَسَنَةِ، وَكَانَ شَدِيد الوَرع وَالزُّهد، تَارِكًا لِجَمِيع مَلَاذ الدُّنيا مِنَ المَآكل، إِلَّا مَا يَأتِيهِ بِهِ أَبُوهُ مِن كَعكٍ وَتِينٍ، وَكَانَ يَلبَس الثِّيَابَ الرثَّة المُرَقَّعة، وَلَا يَدخُل حَمَّامًا، تَرَكَ الفَوَاكِه جَمِيعهَا، وَلَم يَتَنَاوَل مِن الجِهَات.

قُلتُ: وَكَانَ تُؤثر عَنهُ كَرَامَات وَأَحوَال، وَكَانَ أَسمَرَ، كَثَّ اللِّحيَة، رَبعَةً مَهِيبًا، لَا يَرَى الجِدَال وَلَا تُعجِبَهُ المُغَالَبَة، وَيَتَأَذَّى مِمَّن يُجَادِل، وَيَعرِضُ عَنهُ، وَقَلَمُهُ أَبسَط مِن عِبَارَتِهِ، رَحِمَهُ اللهُ تَعَالَى، فَقَد كَانَ عَدِيم النَّظِير.



އެއީ އައްނަވާވީއެވެ. އައްޝައިޚު، އަލްއިމާމު، އަލްގުދްވަތު (އެބަހީ: ނަމޫނާ)، އަލްހާފިޒު، އައްޒާހިދު، އަލްޢާބިދު، އަލްފަގީހު، އަލްމުޖްތަހިދުއް ރައްބާނީއެވެ. ތިމަން ދެކޭ ގޮތުގައި އެއީ ޝައިޚުލް އިސްލާމެކެވެ. އެއީ އިމާމު މުޙްޔިއްދީން، އަބޫ ޒަކަރިއްޔާ، ޔަޙްޔާ ބިން ޝަރަފު ބިން މުއްރީ ބިން ޙަސަން ބިން ޙުސައިން ބިން މުޙައްމަދު ބިން ޙިޒާމު އަލްޙިޒާމީ އަލްޙައުރާނީ އައްޝާފިޢީއެވެ.

އެތަކެއް ފޮތެއް ލިޔުއްވި މުސައްނިފެކެވެ. ދަތުރުވެރިން ވަނީ އެ ފޮތްތައް ދުނިޔޭގެ ހުރިހާ ކަންކޮޅުތަކަށް ފަތުރާފައެވެ. އަދި ބިމުގެ އެންމެ ދުރު ހިސާބުތަކުގައި ވެސް އެ ފޮތްތައް މަޝްހޫރުވެފައި ވެއެވެ.

އެކަލޭގެފާނު އުފަންވެވަޑައިގަތީ ހިޖުރައިން 631 ވަނަ އަހަރުގެ މުޙައްރަމު މަހު، (ސީރިޔާގެ ދިމިޝްގު ކައިރީގައި އޮންނަ) ނަވާ އޭ ކިޔުނު ރަށަށެވެ. އެކަލޭގެފާނުގެ ބައްޕާފުޅު އުޅުއްވީ އެ ރަށުގައި ފިހާރައެއް ހިންގަވާށެވެ. އެކަލޭގެފާނު ބޮޑުފުޅުވެގެން އައީ ރަނގަޅު ރައްކާތެރި މާހައުލެއްގައެވެ. އަދި ގުރްއާން ހިތުދަސް ކުރައްވައިގެން، ބައްޕާފުޅަށް ޓަކައި ފިހާރާގައި މަސައްކަތް ކުރައްވާ އުޅުއްވިއެވެ. އެއަށްފަހު އެކަލޭގެފާނުގެ ބައްޕާފުޅު، އެކަލޭގެފާނު ކިޔަވާވިދާޅުވުމަށް ޓަކައި ދިމިޝްގަށް ގެންދެވިއެވެ. އެކަލޭގެފާނު އުޅުއްވީ ރަވާގިޔާ އޭ ކިޔުނު ހިސާބެއްގައެވެ. ފަރީއްކުޅުވީ (އެ ތަނުގެ މަދަރުސާއިން ކޮންމެ ދުވަހަކު ކިޔަވާ ކުދިންނަށް ދިން) ރޮށިކޮޅެކެވެ. (ޝާފިޢީ ފިގްހުގައި އައްޝީރާޒީ ލިޔުއްވި ފޮތެއް ކަމަށްވާ) އައްތަންބީހު އެކަލޭގެފާނު ކިޔަވާ ވިދާޅުވެ، ހަތަރު މަހާއި ބައިގެ ތެރޭގައި ހިތުދަސް ކުރެއްވިއެވެ. ދެން އެ އަހަރުގެ ބާކީ އޮތް މުއްދަތު، (ޝާފިޢީ ފިގްހުގައި އައްޝީރާޒީ ލިޔުއްވި އެހެން ފޮތެއް ކަމަށްވާ) އަލްމުހައްޛަބުގެ ހަތަރުބައި ކުޅަ އެއްބައި (ހިތުދަސް ކުރައްވައިގެން) ޝައިޚުލް ކަމާލު އިސްޙާގު ބިން އަޙްމަދަށް ކިޔައިދެއްވިއެވެ.

އެއަށްފަހު ބައްޕާފުޅާއެކު އެކަލޭގެފާނު ހައްޖަށް ވަޑައިގަތެވެ. ވިސްނުންފުޅު ތޫނުކަމާއި ފަހުމްވުމުގެ އަލާމާތްތައް އެކަލޭގެފާނުގެ ކިބައިން ފާޅުވިކަން ކަށަވަރެވެ. ކަން ދިމާކުރި ގޮތުން، ނަބިއްޔާގެ ރަށްކޮޅުގައި ދޮޅު މަސް ވަންދެން އެކަލޭގެފާނު މަޑުކުރެއްވިއެވެ. އަދި މަގުމަތީގައި ހިންގަވާ އުޅުއްވި އިރުވެސް، ރާސްތާގެ ބޮޑުބައި ހޭދަ ކުރެއްވީ ކިޔަވާ ވިދާޅުވުމުގައެވެ. ދެން (ދިމިޝްގަށް) އެނބުރިވަޑައިގެން، ދުވާލާއި ރޭގަނޑުގެ ވަގުތުތަކުގައި މަޝްޣޫލުވެ އުޅުއްވީ އިލްމު އުނގެނި ވަޑައިގަތުމުގައެވެ. ފަހެ އެކަލޭގެފާނު ވެވަޑައިގެންނެވީ، (އެކަލޭގެފާނުގެ ދިރިއުޅުމުން) ނަމޫނާ ނެގޭފަދަ ބޭކަލަކަށެވެ. ނުހަނު ބޮޑަށް އަވަހާރަފުޅު އައުމުން އެހެން ގޮތެއް ނެތިގެން މެނުވީ އަވަހާރަފުޅު ނުލައްޕަވައެވެ. ވަގުތު ހޭދަ ކުރެއްވީ ހަމައެކަނި ދަރުސްތަކަށް ހާޒިރުވެ ވަޑައިގަތުމަށާއި، ލިޔުއްވުމަށާއި، ކިޔުއްވުމަށާއި، ޝައިޚުންގެ އަރިހަށް ވަޑައިގަތުމަށެވެ. އެންމެހައި އަރާމުތަކާއި ފަސޭހަތަކެއް ދޫކޮށްލެއްވިއެވެ. އެކަލޭގެފާނު ދިރިއުޅުއްވީ ތަގުވާވެރި، ވަރާވެރި، އަދި ލިބުނު އެއްޗަކާމެދު ހިތްޕުޅު ހަމަޖެހިގެންވާ ގޮތުގައި، ސިއްރާއި ފާޅުގައި ﷲ ބައްލަވާ ވޮޑިގެންވާކަން ރަނގަޅަށް ދެނެވަޑައިގެންވާ ހާލުގައެވެ. ނަފްސުގެ އެދުންތައް ކަމުގައިވާ ރީތި މޮޅު ފޭރާމާއި، ރަނގަޅު މޮޅު ކާތަކެއްޗާއި، ބުރުސޫރަ ރީތިކުރުން ދޫކޮށްލެއްވިއެވެ. ފަރިއްކުޅުވާ އުޅުއްވީ ހަރު ހިކި ރޮށިކޮޅެކެވެ. އަދި ތުރުކުރެއްވީ އެންމެ އާދައިގެ ފޭރާމާއި، ތުނި ފަގުޑިކޮޅެކެވެ. ފަހެ އެކަލޭގެފާނަށް ﷲ ރަހުމަތާއި ރުއްސުން ލައްވައި، އެކަލޭގެފާނުގެ އިލްމަށް ހެޔޮ ޖަޒާ ދެއްވާށިއެވެ.

އެކަލޭގެފާނުގެ އެކުވެރިޔާ، ޝައިޚު އަބުލް ޙަސަން ޢަލީ އިބްނުލް ޢައްޠާރު ވިދާޅުވިއެވެ: ޝައިޚު މުޙްޔިއްދީން (އެބަހީ ނަވަވީ) އެކަލޭގެފާނަށް ވިދާޅުވިއެވެ: ކޮންމެ ދުވަހަކުވެސް ބާރަ ދަރުސް (ފިލާވަޅު) އެކަލޭގެފާނުގެ ޝައިޚުންގެ އަރިހުން އެކަލޭގެފާނު ނެންގެވިއެވެ. އެއީ ޝަރަހައާއި ކުށްފިލުވުމާ އެކުގައެވެ. އެ ގޮތުން ނެންގެވީ (އަލްޣަޒާލީ ލިޔުއްވި ޝާފިޢީ ފިގްހުގެ ފޮތެއް ކަމަށްވާ) އަލްވަސީޠުގައި ދަރުހެކެވެ. އަދި އަލްމުހައްޛަބުގައި ވެސް ދަރުހެކެވެ. އަދި (އަލްޙުމައިދީ ލިޔުއްވި، ސަހީހު ބުޚާރީއާއި މުސްލިމުގެ ޙަދީޘްތައް އެކުލަވާލާފައިވާ ފޮތެއް ކަމަށްވާ) އަލްޖަމްޢު ބައިނައް ޞަޙީޙައިންގައި ވެސް ދަރުހެކެވެ. އަދި ސަހީހު މުސްލިމުގައި ވެސް ދަރުހެކެވެ. އަދި (އަރަބި ބަސް އުނގަންނައިދިނުމަށް) އިބްނުލް ޖިއްނީ ލިޔުއްވި އަލްލުމަޢުގައި ވެސް ދަރުހެކެވެ. އަދި (އަރަބި ބަހުގެ) ޞަރްފުގައި ވެސް ދަރުހެކެވެ. އަދި އުސޫލުލް ފިގްހުގައި ވެސް ދަރުހެކެވެ. އަދި (ޙަދީޘް ރާވީންގެ ވަނަވަރުތަކުގެ އިލްމު ކަމަށްވާ) އަސްމާއުއް ރިޖާލުގައި ވެސް ދަރުހެކެވެ. އަދި (ދީނުގެ އަސާސީ ކަންތައްތަކާއި އަގީދާގެ އިލްމު ކަމަށްވާ) އުސޫލުއް ދީނުގައި ވެސް ދަރުހެކެވެ.

އެކަލޭގެފާނު ވިދާޅުވިއެވެ: އަދި އެ ދަރުސްތަކާ ގުޅުން ހުރި ހުރިހާ ކަމެއް އަހަރެން ލިޔެ ހެދީމެވެ. އެ ގޮތުން އުނދަގޫ ތަންތައްކޮޅު ޝަރަހަކޮށް، އިބާރާތްތައް ބަޔާންކޮށް، ބަސް ރަނގަޅަށް ލިޔުނީމެވެ. އަދި އަހަރެންގެ ވަގުތުގައި ﷲ ބަރަކާތް ލެއްވިއެވެ. އަދި (އެއް ފަހަރަކު) ބޭހުގެ އިލްމައިގެން މަޝްޣޫލުވާން ހިތަށް އަރައިގެން، (އިބްނު ސީނާގެ) އަލްގާނޫން މި ފޮތް އަހަރެން ގަތީމެވެ. ދެން ފަހެ އަހަރެން ހިތް އަނދިރި ވެއްޖެއެވެ. އަދި ދުވަސްތަކެއް ވަންދެން (އެއްވެސް ކަމަކާ) އުޅެން ކުޅަދާނަ ނުވެފައި އަހަރެން ހުރީމެވެ. ދެން އަހަރެންގެ އަމިއްލަ ނަފްސު ހޭވެރިކަމަށް ގެނެސް، އެ (ބޭސް އިލްމުގެ ފޮތް) އަލްގާނޫން ވިއްކާލީމެވެ. ދެން ފަހެ އަހަރެންގެ ހިތް އަލިވެއްޖެއެވެ.

ދެން އެކަލޭގެފާނު އަބަދު ލާޒިމުވެ އުޅުއްވީ، (ދީނުގެ އެކިއެކި އިލްމުތައް އުނގެނިވަޑައިގަތުމުގައި) މަޝްޣޫލުވެ އުޅުއްވުމާއި، ލިޔުއްވުމާއި، ފައިދާ ހާސިލު ކުރެއްވުމައިގެންނެވެ. އެ ކަމުގައި ސަވާބަށް އެދިވަޑައިގެން، ﷲ ގެ ވަޖްހުފުޅު އަދައިގެންވާ ހާލުގައެވެ. އަދިވެސް އުޅުއްވީ ރޯދަ ހިއްޕެވުމާއި، ދަމުނަމާދު ކުރެއްވުމާއި، ޒިކުރު ކުރެއްވުމާއިގެން މުޅިން އެއްކޮށް (އެ ކަންކަމަށް) ލާޒިމުވެ ވަޑައިގެންނެވެ. އޭގެ އިތުރަކަށް (އެހެން އެއްވެސް ކަމެއް) ނުކުރައްވައެވެ.

އިބްނުލް ޢައްޠާރު ވިދާޅުވިއެވެ: ތިމަންނަގެ ޝައިޚު (އެބަހީ: އައްނަވަވީ) ޒިކުރު ކުރެއްވި ގޮތުގައި، އެކަލޭގެފާނުގެ އެއްވެސް ވަގުތެއް ބޭކާރު ގޮތުގައި ހޭދައެއް ނުކުރައްވައެވެ. ރޭގަނޑާއި ދުވާލު ވިޔަސްމެއެވެ. (ދީނީ ކަންކަމާ) މަޝްޣޫލުވެގެން މެނުވީ ނޫޅުއްވައެވެ. މަގުމަތީގައި އުޅުއްވާ އިރަށް ދާންދެންވެސް މެއެވެ. އަދި އެ ގޮތުގެ މަތީގައި ހަ އަހަރު ވަންދެން އެކަލޭގެފާނު ދެމިހުންނެވިއެވެ. ދެން އެއަށްފަހު (ފޮތް) ލިޔުއްވުމާއި، (މީސްތަކުންނަށް) ފައިދާ ކޮށްދިނުމާއި ނަސޭހަތް ދިނުމާއި، ހައްގު ބަސް ވިދާޅުވުމައިގެން އުޅުއްވިއެވެ.

ތިމަންނަ (އައްޛަހަބީ) ބުނަމެވެ: އެކަލޭގެފާނު ފުރިހަމައަށް (ދީނީ) އިލްމަށް ލާޒިމުވެވަޑައިގެން އުޅުއްވައި އަދި އޭގެ މަތީ ދެމިހުންނެވުމާ އެކުގައިވެސް، އެކަލޭގެފާނަކީ އިންތިހާއަށް ވަރާވެރި ބޭކަލެކެވެ. އަދި ހަވާނަފްސުގެ ދަށުދަރަޖައިގެ ކަންކަމާއި، ނުބައި އަޚްލާގުން ނަފްސު ތާހިރުވެ ހުންނެވި ބޭކަލެކެވެ. އަދި ޙަދީޘް ދެނެވަޑައިގަތް، އެ އިލްމުގެ އެކި ފަންނުތަކުގެ މަތީ ގާއިމުވެ ހުންނެވި، އޭގެ ރާވީންގެ ދެނެވަޑައިގަތް، އަދި މަޒުހަބު ފޯރުކޮށްދިނުމުގައި އިސްވެ ހުންނެވި، ނުހަނު ފުންކޮށް އިސްލާމް ދީނުގެ އިލްމުތައް ދެނެވަޑައިގަތް ބޭކަލެކެވެ.

ތިމަންމެންގެ ޝައިޚު އައްރަޝީދު އަލްޙަނަފީ އިބްނުލް މުޢައްލިމު ވިދާޅުވިއެވެ: ޝައިޚު މުޙްޔިއްދީން (އައްނަވަވީ، އެހާ މަދުން ފަރީއްކުޅުވުމާއި ގުޅިގެން) އެކަލޭގެފާނު ފާހަނައަށް ވަޑައިނުގަންނަވާތީ އަދި ދަތި އުނދަގޫކޮށް ދިރުއުޅުއްވާތީ، އެ ކަމާ ނުރުހިގެން އެކަލޭގެފާނު އަދަބުވެރި ކުރުމަށް ތިމަން ވާހަކަ ދެއްކީމެވެ. އަދި އެކަލޭގެފާނު އިލްމުން މެދުކަނޑާލާނެ ބަލިމަޑުކަމެއް ޖެހިދާނެ ކަމަށް ބިރުގެންނެވީމެވެ. ދެންވެސް އެކަލޭގެފާނު ވިދާޅުވިއެވެ: ހަމަކަށަވަރުން މިވެނި މީހަކު (މުސްކުޅިވެގެން ގޮސް) ހަމުގެ ކުލަ ބަދަލުވަންދެން ރޯދަ ހިފައި އުޅުނެވެ.

ޝައިޚު (އައްނަވަވީ) ޖުމްލަ ގޮތެއްގައި މޮޅެތި ކާއެއްޗެހި އަދި މޭވާތައް (އަމިއްލަ ނަފްސަށް) މަނާ ކުރެއްވިއެވެ. އަދި އެކަލޭގެފާނު ވިދާޅުވެއެވެ: އަހަރެން ބަނޑުފުރިގެން ނިދިއައިސްދާނެ ކަމަށް ބިރުގަންނަމެވެ. އަދި އާންމު ގޮތެއްގައި ދުވާލާއި ރޭގަނޑު އެކަލޭގެފާނު ފަރީއްކުޅުވަނީ އެންމެ ކާއެތިކޮޅެކެވެ. އެއަށްފަހު (ފަތިސްވުމުގެ ކުރިން) ރޭގަނޑުގެ ފަހުކޮޅުގައި އެއް ފަހަރު (ފެން) ހިއްޕަވައެވެ.

އިބްނުލް ޢައްޠާރު ވިދާޅުވިއެވެ: އެކަލޭގެފާނު މޭވާ (ފަރީއްނުކުޅުވާތީ، އެ ކަމާ) ބެހޭ ގޮތުން ތިމަންނަ ވާހަކަދެއްކީމެވެ. ފަހެ އެކަލޭގެފާނު ވިދާޅުވިއެވެ: ދިމިޝްގަކީ ވަރަށް ގިނަ ވަގްފު ބިންތަކާއި، (މަސްލަހަތަށް ޓަކައި) ބޭނުން ކުރެވިގެން ނުވާނެ (ޔަތީމުންނާއި ކުޑަކުދިންނާއި ބުއްދިފިލާފައިވާ މީހުން ފަދަ ފަރާތްތަކުގެ) މުދާތައް ގިނަ ރަށެކެވެ. (އެބަހީ: އެ މުދާތައް ބަލަހައްޓަން ޖެހޭ ހައްގު ގޮތުގައި މީހުން ނުބަލަހައްޓާތީ، ލިބޭ މޭވާއަކީ ހަލާލު ގޮތުގައި އައިސްފައިވާ އެއްޗަކަށް ނުވެދާނެއެވެ.) ދެން އަނެއް ކަމަކީ، އެ (ދަނޑުބިންތައް ގެނގުޅުމުގައި މީހުން) ބޭނުން ކުރަނީ މުސާގާގެ މުއާމަލާތެވެ. (އެބަހީ: މަސައްކަތު މީހެއް އެ ދަނޑުބިންތަކުގައި ފެންދީ ގަސްހައްދާ މަތިން، އެ ބިމުން ހެއްދޭ ތަކެތިން ބައެއް އަގުގެ ގޮތުގައި ދިނުމަށް ބިމުގެ ވެރިޔާ އެއްބަސްވުމެވެ.) އަދި އެ މުއާމަލާތް (ހުއްދަކަމާމެދު) ޚިލާފު އޮވެއެވެ. ވީމާ އެ (މޭވާތައް) ކެއުމާމެދު އަހަރެން ހިތްހަމަޖެހޭނީ ކިހިނަކުންހެއްޔެވެ؟

އަދި ނާދިރު ފަހަރަކު މެނުވީ އެކަލޭގެފާނު އެއްވެސް މީހެއްގެ އަތުން އެއްޗެއް (ހަދިޔާއެއްގެ ގޮތުގައި) ބައްލަވައި ނުގަންނަވާއެވެ. (ނޯޓު: އެއީ އޭގެ ހަލާލުކަމާމެދު ޝައްކުވާތީ، ނުވަތަ އެކަލޭގެފާނުގެ ކިބައިން ކަމެއް އެދިގެން ދޭ އެއްޗަކަށް ވެދާނެތީއެވެ. ނުވަތަ މޮޅު އެއްޗެހި ބޭނުން ނުވާތީއެވެ.) އަދި އެކަލޭގެފާނު މަޝްޣޫލު ނުކުރުވާނެ މީހެއްގެ ފަރާތުން ކުޑަ އެއްޗެއް އެކަލޭގެފާނު ބައްލަވައިގަނެއެވެ. އެއްފަހަރަކު ފަގީރަކު އެކަލޭގެފާނަށް (ފެން އަޅާ) ކަންވާރެއް ހަދިޔާ ކުރެއްވުމުން އެކަލޭގެފާނު އެ ބައްލަވައިގަތެވެ. އަދި (އެއް ފަހަރަކު) އެކަލޭގެފާނުގެ އެކުވެރިޔާ އަލްޚަޠީބު ބުރްހާނުއްދީން އަލްއިސްކަންދަރާނީ، އެކަލޭގެފާނާއެކު ރޯދާވީއްލަން އެއްބަސްކުރުވިއެވެ. އަދި (ބުރްހާނުއްދީން) ވިދާޅުވިއެވެ: ކާތަކެތި ގެންނަވާށެވެ. އަދި ދެ މީހުން އެކީގައި ރޯދަވީއްލާނީއެވެ. ދެން ބުރްހާނުއްދީން އެއިން ފަރީއްކުޅުވި އިރު، އެއީ (އެންމެ) ދެ ވައްތަރެއްގެ ކާތަކެތި ކަމުގައި ވިއެވެ. އަދި (ރޮއްޓާ ޖައްސައިގެން ކެއުމަށް) ދެ ވައްތަރެއްގެ ދިޔަ (އެއްފަހަރާ) އައްނަވަވީ ފަރީއްކުޅުވަނީ މަދު ފަހަރެއްގައެވެ. އަދި އެކަލޭގެފާނަކީ ހިނިފުޅުވެޑުވުން މަދު، ސަމާސާ ނުކުރައްވާ ބޭކަލެކެވެ. އަދި ކިއެއްތަ، އެކަލޭގެފާނު އަވަހަށް ވަކިވެވަޑައިގަންނަވައެވެ. އެކަލޭގެފާނު ހައްގު ބަސް ވިދާޅުވެއެވެ. އެކަލޭގެފާނާ ދެކޮޅަށްވި ކަމުގައި ވިޔަސްމެއެވެ. ﷲ ގެ މަގުގައި އެކަލޭގެފާނު މީހަކު ކުރާ މަލާމާތަކަށް ނުބައްލަވައެވެ. އަދި (ނުބައި) ވެރިންނާއި އަނިޔާވަރިކަމަށް އެކަލޭގެފާނު އިންކާރުން ކުރިމަތިލައްވައެވެ. އަދި އެބައިމީހުންނަށް ލިޔެ (ނަސޭހަތް ދީ) ﷲ އަށް ބިރުވެތި ކުރައްވައެވެ.

އިބްނު ފަރްޙި ވިދާޅުވިއެވެ. -އަދި އެކަލޭގެފާނަކީ ޝައިޚު (އައްނަވަވީގެ ފޮތްތަކުގެ) މައްޗަށް ޝަރަހަ ދެއްވި ބޭކަލެކެވެ.- ޝައިޚު މުޙްޔިއްދީން (އެބަހީ: އައްނަވަވީ) ވެވަޑައިގެންނެވީ އިލްމާއި، ޒުހުދާއި، ހެޔޮކަންކަމަށް އަމުރުކޮށް، ނުބައިކަންކަން މަނާކުރުމުގައި، އެ ދަރަޖައަށް ވާސިލުވެވިއްޖެ ނަމަ އެ ބޭކަލެއްގެ ގާތަށް (އިލްމު އުނގެނުމަށް) ދަތުރުހަދާފައި ދާ ދަރަޖައިގެ ބޭކަލަކަށެވެ.

އަދި ޝައިޚު (އައްނަވަވީ) އުޅުއްވީ މަދު އެތިކޮޅަކުން ފުއްދަވާލައްވައިގެންނެވެ. ﷲ އެކަލޭފެފާނަށް ރުއްސުން ލައްވާށިއެވެ. އަދި އެކަލޭގެފާނުގެ އުމުރުފުޅުގެ ހަގުކަމާއި، އެކަލޭގެފާނުގެ ޝައިޚުން ދިރިހުންނަވަނިކޮށް (އެ ބޭކަލުންގެ ސަނަދުތައް މަތިވެފައި) އެކަލޭގެފާނުގެ ރިވާޔަތްތަކުގެ ސަނަދު ތިރިވުމާއެކުގައި ވެސް، އިމާމު އަބޫ ޝާމާހަށް ފަހު (ދިމިޝްގައި ޙަދީޘް އިލްމުގެ މަރުކަޒެއް ކަމަށްވާ) ދާރުލް ޙަދީޘް އަލްއަޝްރަފިއްޔާ އާ އެކަލޭގެފާނު ހަވާލުވެވަޑައިގަތެވެ. ފަހެ ތިމަންނައަށް ފޯރާފައިވާ ޚަބަރުތަކުން، އެކަލޭގެފާނަށް ތަނަވަސްކަމާއި ބާރުވެރިކަން ލިއްބައިދޭނޭ ކަމެއް ތިމަންނައަކަށް ނުފެނުނެވެ. އަދި ކިއެއް، އެކަލޭގެފާނަށް ލިބިވަޑައިގަންނަވަނީ، އެކަލޭގެފާނުގެ ބައްޕާފުޅު ފަރާތުން ފަރީއްކުޅުވާނެ އެއްޗެކެވެ. އަދި އެކަލޭގެފާނަށް ލިބިވަޑައިގަތް މުސާރަކޮޅުން ބައްލަވައިގަތީ ފިގްހާއި (އެހެނިހެން) ފޮތްތަކެވެ. އެކަލޭގެފާނު ދަތުރުކޮށް ބައިތުލް މަގްދިސަށް ޒިޔާރާތް ކުރެއްވިއެވެ. އަދި އަނބުރާ (އުފަންރަށް) ނަވާއަށް ވަޑައިގަތީ އާލާސްކަންފުޅުގައެވެ. އަދި އެކަލޭގެފާނު އަވަހާރަވެ ﷲ ގެ ހަޒްރަތަށް ރުޖޫއަވެވަޑައިގަތީ، ހިޖުރައިން 676 ވަނަ އަހަރުގެ ރަޖަބު މަހުގެ 24 ވަނަ ދުވަހުގައެވެ.

ތިމަންމެންގެ ޝައިޚު، ގުޠުބުއްދީން މޫސާ ވިދާޅުވިއެވެ: (އައްނަވަވީއަކީ) އެކަލޭގެފާނުގެ ޒަމާނުގައި އިލްމާއި، ވަރާވެރިކަމާއި، އަޅުވެރިކަމާއި، މަދު އެތިކޮޅަކުން ފުއްދާލުމާއި، ދަތި ދިރިއުޅުއްވުމުގައި ލާމަސީލު ބޭކަލެކެވެ. ކޯޓުގައި އެކަލޭގެފާނު އެއްފަހަރަށް ވުރެ ގިނައިން ރަސްގެފާނު އައްޡާހިރާ ކުރިމަތިލެއްވިއެވެ. ފަހެ ބުނެވޭ ގޮތުގައި އެ ރަސްގެފާނުގެ ވިދާޅުވިއެވެ: މި ބޭކަލެއްގެ ކިބައިން ތިމަންނަ ބިރުގަންނަމެވެ. އަދި އަލްފަގީހު ޝަމްސުއްދީން މުޙައްމަދު ބިން ފަޚްރު ވިދާޅުވިއެވެ: އެކަލޭގެފާނަކީ އިމާމުވެރި، މޮޅު ކުޅަދާނަ، (ގިނަގުނަ އިލްމުތައް ރައްކާތެރިކޮށް) ހިފްޒުކުރެއްވި، ފަތުވާ ދެއްވި ބޭކަލެކެވެ. އެތައް ތަފާތު އިލްމެއް އެންމެ ފުރިހަމައަށް އުނގެނިވަޑައިގަތެވެ. އަދި ރަނގަޅު އެތައް ފޮތެއް ލިޔުއްވިއެވެ. އަދި އެކަލޭގެފާނަކީ އިންތިހާއަށް ވަރާވެރި ޒުހުދުވެރި ބޭކަލެކެވެ. ކާތަކެތީގެ ދުނިޔޭގެ ހުރިހާ ލައްޒަތްތައް ދޫކުރެއްވިއެވެ. އެއީ އެކަލޭގެފާނުގެ ބައްޕާފުޅުގެ ފަރާތުން އެކަލޭގެފާނާ ހަމައަށް އަންނަ (ރޮށީގެ ވައްތަރެއް ކަމަށްވާ) ކޭކާއި ވައްތީނިކޮޅެއް މެނުވީއެވެ. އަދި ލައި އުޅުއްވީ ރިވެތި ނޫން، ފޮތިއަޅާފައިވާ ފޭރާމެވެ. އަދި (އެހާ މަދުން ފަރީއްކުޅުވުމުގެ ސަބަބުން) ފާހަނައަށް ވަދެވަޑައި ނުގަންނަވައެވެ. ހުރިހާ ވައްތަރެއްގެ މޭވާ ދޫކުރެއްވިއެވެ. އަދި (ކާތަށީގެ) އެކި ކޮޅުކޮޅުން އެކަލޭގެފާނު ފަރީއްނުކުޅުވައެވެ.

ތިމަންނަ (އައްޛަހަބީ) ބުނަމެވެ: އެކަލޭގެފާނާ ގުޅިގެން ކަރާމާތްތަކާއި (އަޖައިބުކުރުވަނި) ހާލަތްތައް އައިސްފައިވެއެވެ. އަދި އެކަލޭގެފާނަކީ މުށިކުލައިގެ ހަންކޮޅެއް ލިބިގެންވާ، ތުނބުޅިފުޅު ބޯ، ބައްޓަން މެދުމިން، ހައިބަތު ހުންނެވި ބޭކަލެކެވެ. ޖަދަލު ކުރުން އެކަލޭގެފާނު ބައްލަވައެއް ނުގަނެއެވެ. އަދި (އެކަކު އަނެކަކު) ބަލިކުރަން (ވާހަކަދެއްކުން) އެކަލޭގެފާނަށް ހިއްޕުޅާނުވެއެވެ. އަދި ޖަދަލު ކުރާ މީހުންގެ ފަރާތުން އެކަލޭގެފާނަށް އުނދަގޫފުޅުވެއެވެ. އަދި އެ މީހެއްގެ ކިބައިން އެކަލޭގެފާނު މޫނު އަނބުރާލައްވައެވެ. އަދި އެކަލޭގެފާނުގެ (ވާހަކަ ދެއްކެވުމުގެ) އިބާރާތްތަކަށް ވުރެ، އެކަލޭގެފާނުގެ ގަލަމުގެ (އިބާރާތްތައް) ފުޅާވެގެންވެއެވެ. ﷲ ތަޢާލާ އެކަލޭގެފާނަށް ރަހުމަތް ލައްވާށިއެވެ. ހަމަކަށަވަރުން އެކަލޭގެފާނަކީ އެފަދަ މިސާލެއް ނުވާ ބޭކަލެކެވެ.



— އަނބުރާ މައި ސަފުހާއަށް —
